% --------------------------------------------------------
% 1. 模型定义
% --------------------------------------------------------
\begin{modelbox}[模型：一维量子谐振子]
\textbf{哈密顿量}
\[
\hat{H} = \frac{\hat{P}^2}{2m} + \frac{1}{2}m\omega^2 \hat{X}^2
= \hbar\omega\Bigl(\hat{a}^\dagger \hat{a} + \frac{1}{2}\Bigr).
\]

\textbf{特征尺度}
\begin{itemize}[nosep]
    \item 特征长度 $x_0 = \sqrt{\hbar/m\omega}$，自然坐标 $\xi = x/x_0$
    \item 特征动量 $p_0 = \sqrt{m\omega\hbar}$
\end{itemize}

\textbf{升降算符}
\[
\hat{a} = \frac{1}{\sqrt{2}}\Bigl(\xi + \frac{\partial}{\partial\xi}\Bigr),
\qquad
\hat{a}^\dagger = \frac{1}{\sqrt{2}}\Bigl(\xi - \frac{\partial}{\partial\xi}\Bigr),
\qquad [\hat{a}, \hat{a}^\dagger] = 1.
\]

\textbf{位置与动量}
\[
\hat{X} = x_0\,\frac{\hat{a} + \hat{a}^\dagger}{\sqrt{2}},
\qquad
\hat{P} = \frac{p_0}{i}\,\frac{\hat{a} - \hat{a}^\dagger}{\sqrt{2}}.
\]
\end{modelbox}

% --------------------------------------------------------
% 2. 代数解：能谱与本征态
% --------------------------------------------------------
\subsection{代数解：能谱与本征态}

谐振子是少数几个可以用纯代数方法完全求解的模型。核心观察是：哈密顿量可以写成粒子数算符 $\hat{N}=\hat{a}^\dagger\hat{a}$ 的线性函数。

\begin{derivationbox}[能谱推导]
由对易关系 $[\hat{N}, \hat{a}] = -\hat{a}$ 和 $[\hat{N}, \hat{a}^\dagger] = \hat{a}^\dagger$ 可知：
若 $\hat{N}|n\rangle = n|n\rangle$，则
\[
\hat{N}\bigl(\hat{a}|n\rangle\bigr) = (n-1)\bigl(\hat{a}|n\rangle\bigr),
\qquad
\hat{N}\bigl(\hat{a}^\dagger|n\rangle\bigr) = (n+1)\bigl(\hat{a}^\dagger|n\rangle\bigr).
\]
因此 $\hat{a}$ 将本征态的量子数减 1，$\hat{a}^\dagger$ 加 1。

由于 $\hat{N}$ 是半正定算符，$\langle n|\hat{N}|n\rangle = \|\hat{a}|n\rangle\|^2 \geq 0$，存在基态 $|0\rangle$ 满足
\[
\hat{a}|0\rangle = 0.
\]
用坐标表象写出此方程：$x_0(\xi + \partial_\xi)\psi_0 = 0$，解得高斯波函数
\[
\psi_0(x) = \Bigl(\frac{m\omega}{\pi\hbar}\Bigr)^{1/4} e^{-\xi^2/2}.
\]
反复作用 $\hat{a}^\dagger$ 可得任意激发态。能谱为
\begin{equation}
\boxed{E_n = \hbar\omega\Bigl(n + \frac{1}{2}\Bigr),\quad n = 0,1,2,\dots}
\end{equation}
\end{derivationbox}

能级等间距分布，间隔 $\hbar\omega$。这是谐振子最重要的谱特征，也是后续所有推导（相干态、求和规则、WKB验证）的共同起点。

% --------------------------------------------------------
% 3. 波函数与 Hermite 多项式
% --------------------------------------------------------
\subsection{波函数与 Hermite 多项式}

坐标表象下的定态波函数为
\begin{equation}
\psi_n(x) = \frac{1}{\sqrt{2^n n!}}\Bigl(\frac{m\omega}{\pi\hbar}\Bigr)^{1/4}
H_n(\xi)\,e^{-\xi^2/2},
\qquad \xi = \frac{x}{x_0}.
\end{equation}
其中 $H_n(\xi)$ 为 Hermite 多项式，满足递推关系
\[
H_{n+1}(\xi) = 2\xi H_n(\xi) - 2n H_{n-1}(\xi),
\qquad H_n'(\xi) = 2n H_{n-1}(\xi).
\]

\begin{insightbox}[波函数的结构]
谐振子波函数是\textbf{高斯包络}与\textbf{多项式振荡}的乘积。
\begin{itemize}[nosep]
    \item 基态 $|0\rangle$：纯高斯，无节点，最小不确定波包
    \item 激发态 $|n\rangle$：$n$ 个节点，经典转折点外呈指数衰减，转折点内振荡
    \item 宇称：$H_n(-\xi) = (-1)^n H_n(\xi)$，故 $\psi_n(x)$ 具有确定宇称 $(-1)^n$
\end{itemize}
经典转折点位于 $x_\text{cl} = \pm x_0\sqrt{2n+1}$。当 $n\gg 1$ 时，WKB 近似在该区域给出与精确波函数一致的渐近行为。
\end{insightbox}

\subsubsection{前几个波函数的具体形式}

用递推关系可直接写出前几个归一化波函数：
\begin{align*}
\psi_0(x) &= \Bigl(\frac{m\omega}{\pi\hbar}\Bigr)^{1/4} e^{-\xi^2/2}, \\
\psi_1(x) &= \Bigl(\frac{m\omega}{\pi\hbar}\Bigr)^{1/4} \sqrt{2}\,\xi\,e^{-\xi^2/2}, \\
\psi_2(x) &= \Bigl(\frac{m\omega}{\pi\hbar}\Bigr)^{1/4} \frac{1}{\sqrt{2}}(2\xi^2-1)\,e^{-\xi^2/2}, \\
\psi_3(x) &= \Bigl(\frac{m\omega}{\pi\hbar}\Bigr)^{1/4} \frac{1}{\sqrt{3}}(2\xi^3-3\xi)\,e^{-\xi^2/2}.
\end{align*}

\begin{tipbox}[速记特征]
\begin{itemize}[nosep]
    \item $|0\rangle$：单峰，中心最大，$\psi_0(0)$ 取最大值
    \item $|1\rangle$：奇函数，中心节点，两侧各一峰
    \item $|2\rangle$：偶函数，两侧各一节点，中心为峰
    \item $|3\rangle$：奇函数，中心峰两侧各一节点，外侧两峰
\end{itemize}
所有波函数在 $|x|\gg x_0$ 时均按 $e^{-\xi^2/2}$ 指数衰减。这些简单形式在考场中可直接写出，无需背诵一般 Hermite 多项式。
\end{tipbox}

\subsection{能量表象下的算符结构}

在粒子数基 $\{|n\rangle\}$ 下，升降算符的矩阵元极其简单：
\begin{equation}
\boxed{\langle n'|\hat{a}|n\rangle = \sqrt{n}\,\delta_{n',n-1},
\qquad
\langle n'|\hat{a}^\dagger|n\rangle = \sqrt{n+1}\,\delta_{n',n+1}.}
\end{equation}

由此可得位置与动量的矩阵元。由于 $\hat{X}\propto \hat{a}+\hat{a}^\dagger$，它只连接相邻能级：
\begin{align}
\langle n'|\hat{X}|n\rangle &= x_0\sqrt{\frac{n+1}{2}}\,\delta_{n',n+1}
+ x_0\sqrt{\frac{n}{2}}\,\delta_{n',n-1}, \\
\langle n'|\hat{P}|n\rangle &= \frac{p_0}{i}\sqrt{\frac{n+1}{2}}\,\delta_{n',n+1}
- \frac{p_0}{i}\sqrt{\frac{n}{2}}\,\delta_{n',n-1}.
\end{align}

$\hat{X}^2$ 包含 $(\hat{a}+\hat{a}^\dagger)^2 = \hat{a}^2 + (\hat{a}^\dagger)^2 + 2\hat{a}^\dagger\hat{a} + 1$，因此除对角元外还连接 $|n\rangle$ 与 $|n\pm 2\rangle$：
\begin{equation}
\langle n'|\hat{X}^2|n\rangle = \frac{x_0^2}{2}\Bigl[(2n+1)\delta_{n',n}
+ \sqrt{(n+1)(n+2)}\,\delta_{n',n+2}
+ \sqrt{n(n-1)}\,\delta_{n',n-2}\Bigr].
\end{equation}

\begin{tipbox}[矩阵元的选择定则]
\begin{itemize}[nosep]
    \item $\hat{X}$：$\Delta n = \pm 1$（电偶极跃迁选择定则）
    \item $\hat{X}^2$：$\Delta n = 0, \pm 2$
    \item $\hat{P}$：与 $\hat{X}$ 同结构，仅系数差一个因子 $i$
\end{itemize}
这些选择定则直接决定了谐振子在外场中的响应行为：线性耦合 $x$ 只能诱导相邻能级跃迁，二次耦合 $x^2$ 可诱导同级和隔级跃迁。
\end{tipbox}

由矩阵元立即可得坐标与动量的方差：
\[
\langle n|\hat{X}^2|n\rangle = \frac{x_0^2}{2}(2n+1) = \frac{(n+\tfrac12)\hbar}{m\omega},
\qquad
\langle n|\hat{P}^2|n\rangle = \frac{p_0^2}{2}(2n+1) = (n+\tfrac12)m\omega\hbar.
\]
因此不确定度积为
\[
\Delta X\,\Delta P = \frac{\hbar}{2}(2n+1) \geq \frac{\hbar}{2},
\]
基态达到下限，是标准的"最小不确定态"。

\subsubsection{能量表象的数学基础：投影算符 $A(m,n)$}

能量表象中的矩阵元可以通过投影算符 $A(m,n)\equiv|m\rangle\langle n|$ 系统地表示。在 $H$ 的本征基 $\{|n\rangle\}$ 下，$A(m,n)$ 的矩阵元为
\[
[A(m,n)]_{kl} = \langle k|m\rangle\langle n|l\rangle = \delta_{km}\delta_{nl}.
\]
即第 $m$ 行第 $n$ 列为 1，其余为 0。

\textbf{基本性质}：
\begin{itemize}[nosep]
    \item 厄米共轭：$A^\dagger(m,n) = |n\rangle\langle m| = A(n,m)$
    \item 迹：$\operatorname{Tr}A(m,n) = \sum_k\langle k|m\rangle\langle n|k\rangle = \delta_{mn}$
    \item 乘法：$A(m,n)A(k,l) = \delta_{nk}A(m,l)$
\end{itemize}

\textbf{完备性展开}：任意算符 $\hat{F}$ 可唯一展开为
\begin{equation}
\boxed{\hat{F} = \sum_{m,n}F_{mn}A(m,n),
\qquad F_{mn}=\langle m|\hat{F}|n\rangle = \operatorname{Tr}\bigl(\hat{F}A^\dagger(m,n)\bigr).}
\end{equation}
\textbf{Trace 验证}（以 $F_{mn}$ 的第二种表达式为例）：
\begin{align*}
\operatorname{Tr}\bigl(\hat{F}A^\dagger(m,n)\bigr)
&= \operatorname{Tr}\bigl(\hat{F}|n\rangle\langle m|\bigr)
= \sum_k \langle k|\hat{F}|n\rangle\langle m|k\rangle \\
&= \sum_k \langle m|k\rangle\langle k|\hat{F}|n\rangle
= \langle m|\hat{F}|n\rangle = F_{mn}.
\end{align*}
第一步插入完备性关系 $\sum_k|k\rangle\langle k|=\hat{1}$，第二步利用 $\langle m|k\rangle=\delta_{mk}$ 完成对 $k$ 的求和。

\begin{insightbox}[投影算符与谐振子]
对谐振子，$A(m,n)$ 的显式形式为
\[
A(m,n) = \frac{1}{\sqrt{m!n!}}(\hat{a}^\dagger)^m|0\rangle\langle 0|\hat{a}^n.
\]
利用 $|0\rangle\langle 0| = :e^{-\hat{a}^\dagger\hat{a}}:$（正规序），可将任意算符的矩阵元计算转化为 Wick 定理的应用。这是多体理论中格林函数方法的基础。
\end{insightbox}

\subsection{相干态与位移算符}

升降算符代数不仅给出能谱，还自然引出谐振子的一个核心结构——\textbf{相干态}（coherent state）。

\subsubsection{位移算符的定义}

定义位移算符
\begin{equation}
\hat{D}(\alpha) = \exp(\alpha\hat{a}^\dagger - \alpha^*\hat{a}),
\qquad \alpha\in\mathbb{C}.
\end{equation}
由 Baker--Campbell--Hausdorff 公式及 $[\hat{a},\hat{a}^\dagger]=1$ 可得
\begin{equation}
\hat{D}(\alpha) = e^{-|\alpha|^2/2}\,e^{\alpha\hat{a}^\dagger}\,e^{-\alpha^*\hat{a}}.
\end{equation}
作用在真空态上产生\textbf{相干态} $|\alpha\rangle = \hat{D}(\alpha)|0\rangle$，其在 Fock 基下的展开为
\begin{equation}
\boxed{|\alpha\rangle = e^{-|\alpha|^2/2}\sum_{n=0}^{\infty}\frac{\alpha^n}{\sqrt{n!}}|n\rangle.}
\end{equation}

\subsubsection{位置表象的位移}

将 $\hat{a},\hat{a}^\dagger$ 用 $\hat{X},\hat{P}$ 表示，位移算符可改写为
\[
\hat{D}(\alpha) = \exp\Bigl[\frac{i}{\hbar}\bigl(p_0\,\text{Re}\,\alpha\bigr)\hat{X}
- \frac{i}{\hbar}\bigl(\sqrt{2}\,p_0\,\text{Im}\,\alpha\bigr)\hat{P}\Bigr].
\]

这意味着：
\begin{itemize}[nosep]
    \item 当 $\alpha$ 为\textbf{实数}时，$\hat{D}(\alpha) = \exp(-i p_0 \alpha \hat{X}/\hbar)$，实现\textbf{动量空间的平移}，$\langle\hat{P}\rangle$ 改变而 $\langle\hat{X}\rangle$ 不变
    \item 当 $\alpha$ 为\textbf{纯虚数}时，$\hat{D}(\alpha) = \exp(i \sqrt{2}p_0 \,\text{Im}\alpha \,\hat{P}/\hbar)$，实现\textbf{位置空间的平移}，$\langle\hat{X}\rangle$ 改变而 $\langle\hat{P}\rangle$ 不变
\end{itemize}

\begin{formulabox}[相干态的期望值]
\begin{align*}
\langle\alpha|\hat{X}|\alpha\rangle &= \sqrt{2}\,x_0\,\text{Re}\,\alpha, \\
\langle\alpha|\hat{P}|\alpha\rangle &= \sqrt{2}\,p_0\,\text{Im}\,\alpha, \\
\langle\alpha|\hat{H}|\alpha\rangle &= \hbar\omega\Bigl(|\alpha|^2 + \frac12\Bigr).
\end{align*}
\end{formulabox}

\subsubsection{能量测量概率：泊松分布}

相干态在能量本征基上的投影为
\[
\langle n|\alpha\rangle = e^{-|\alpha|^2/2}\frac{\alpha^n}{\sqrt{n!}},
\]
因此能量测量概率服从\textbf{泊松分布}：
\begin{equation}
\boxed{P_n = |\langle n|\alpha\rangle|^2 = \frac{e^{-|\alpha|^2}|\alpha|^{2n}}{n!}
= \frac{e^{-\bar{n}}\,\bar{n}^{\,n}}{n!},\quad \bar{n}=|\alpha|^2.}
\end{equation}

泊松分布的涨落特征 $\Delta n = \sqrt{\bar{n}}$ 是相干态的标志性性质。平均能量为
\[
\langle E\rangle = \hbar\omega(\bar{n} + \tfrac12)
= \frac{\langle P\rangle^2}{2m} + \frac12 m\omega^2 \langle X\rangle^2 + \frac{\hbar\omega}{2},
\]
其中前两项正是经典谐振子的能量，最后一项是量子零点能。

\begin{insightbox}[相干态的物理图像——为什么波包不扩散？]
相干态是量子谐振子中"最经典"的态。在相空间中，它是一个高斯波包，中心沿经典轨迹 $(\langle X\rangle(t), \langle P\rangle(t))$ 做简谐振荡。

\textbf{关键问题}：自由粒子的高斯波包会扩散（$\Delta X$ 随时间增大），为什么谐振子势中的相干态不扩散？

\textbf{回答：因为谐振子提供了恢复力，把波包"拉回来"；自由粒子没有恢复力，波包一旦展开就持续扩散。}

用 Heisenberg 图像可以精确说明。对任意态，位置方差为
\[
(\Delta X)^2(t) = \langle X^2(t)\rangle - \langle X(t)\rangle^2.
\]

\textbf{自由粒子}（$V=0$）：$X(t) = X + Pt/m$，故
\[
(\Delta X)^2(t) = (\Delta X)^2_0 + \frac{(\Delta P)^2_0}{m^2}t^2 + \frac{\langle XP+PX\rangle_0 - 2\langle X\rangle_0\langle P\rangle_0}{m}t.
\]
其中 $t^2$ 项导致方差\textbf{单调增长}，波包不可逆地扩散。

\textbf{谐振子}：$X(t) = X\cos\omega t + \frac{P}{m\omega}\sin\omega t$，故
\begin{align*}
(\Delta X)^2(t)
&= (\Delta X)^2_0\cos^2\omega t + \frac{(\Delta P)^2_0}{m^2\omega^2}\sin^2\omega t \\
&\quad + \frac{\langle XP+PX\rangle_0 - 2\langle X\rangle_0\langle P\rangle_0}{m\omega}\sin\omega t\cos\omega t.
\end{align*}
方差是 $\cos^2\omega t$、$\sin^2\omega t$ 和 $\sin\omega t\cos\omega t$ 的组合，因此以 $2\omega$ 的频率\textbf{周期性振荡}——波包会周期性地"收缩"回来。

\textbf{相干态的特殊性}：相干态是最小不确定态，且 $X$ 与 $P$ 的不确定部分互不关联，交叉项为零。同时其自然宽度恰好满足
\[
(\Delta X)^2_0 = \frac{\hbar}{2m\omega},
\qquad
\frac{(\Delta P)^2_0}{m^2\omega^2} = \frac{\hbar}{2m\omega} = (\Delta X)^2_0.
\]
因此
\[
(\Delta X)^2(t) = (\Delta X)^2_0(\cos^2\omega t + \sin^2\omega t) = (\Delta X)^2_0.
\]
方差\textbf{严格不随时间变化}——波包形状完全不变，只有中心在振荡。 squeezed 态则对应交叉项非零或 $(\Delta X)_0\neq (\Delta P)_0/(m\omega)$ 的情况，会以 $2\omega$ 频率"呼吸"。
\end{insightbox}
\subsubsection{Heisenberg 图像下的时间演化}

在 Heisenberg 图像中，算符随时间演化而态不变。谐振子的 Heisenberg 运动方程为
\[
\frac{d\hat{X}_H}{dt} = \frac{1}{i\hbar}[\hat{X}, \hat{H}] = \frac{\hat{P}}{m},
\qquad
\frac{d\hat{P}_H}{dt} = \frac{1}{i\hbar}[\hat{P}, \hat{H}] = -m\omega^2\hat{X}.
\]

联立解得
\begin{equation}
\hat{X}_H(t) = \hat{X}\cos\omega t + \frac{\hat{P}}{m\omega}\sin\omega t,
\qquad
\hat{P}_H(t) = \hat{P}\cos\omega t - m\omega\hat{X}\sin\omega t.
\end{equation}

对相干态 $|\alpha\rangle$，期望值直接继承经典轨迹：
\begin{align*}
\langle\hat{X}_H(t)\rangle
&= \langle\hat{X}\rangle_0\cos\omega t + \frac{\langle\hat{P}\rangle_0}{m\omega}\sin\omega t \\
&= \sqrt{2}\,x_0\,\text{Re}(\alpha)\cos\omega t + \sqrt{2}\,x_0\,\text{Im}(\alpha)\sin\omega t \\
&= \sqrt{2}\,x_0\,\text{Re}(\alpha e^{-i\omega t}).
\end{align*}

同理
\[
\langle\hat{P}_H(t)\rangle = \sqrt{2}\,p_0\,\text{Im}(\alpha e^{-i\omega t}).
\]

这恰好是相空间中半径为 $\sqrt{2}x_0|\alpha|$ 的圆轨道，角频率 $\omega$。

% 6. 含时演化：瞬时脉冲与相干态的物理实现
% --------------------------------------------------------
\subsection{含时演化：瞬时脉冲与相干态的物理实现}

相干态不仅是数学构造，还是真实物理过程的产物。考虑谐振子基态受到一个瞬时脉冲：
\[
\hat{H}'(t) = \lambda\,\hat{X}\,\delta(t).
\]

\subsubsection{精确时间演化算符}

对 $\delta(t)$ 型的奇异微扰，不能直接使用含时微扰论。应将薛定谔方程在 $t=0$ 附近积分：
\[
|\psi(0^+)\rangle - |\psi(0^-)\rangle
= \frac{1}{i\hbar}\int_{-\epsilon}^{+\epsilon}\lambda\hat{X}\delta(t')|\psi(t')\rangle\,dt'
= \frac{\lambda\hat{X}}{i\hbar}|\psi(0)\rangle.
\]
因此态矢量在脉冲瞬间发生有限跳跃：
\begin{equation}
|\psi(0^+)\rangle = e^{-i\lambda\hat{X}/\hbar}|\psi(0^-)\rangle.
\end{equation}
时间演化算符是\textbf{精确指数}，不是微扰级数——这是 $\delta(t)$ 脉冲与常微扰、简谐微扰的本质区别。

\subsubsection{t>0 后的演化}

\begin{warnbox}[常见困惑：脉冲之后呢？]
很多考生会问："$\delta(t)$ 只在 $t=0$ 作用，那 $t>0$ 之后态就不演化了吗？"

\textbf{答案是：态仍在演化，但能量测量概率不变。}

$t>0$ 时微扰消失，系统恢复为自由谐振子。$t=0^+$ 时刻的态 $|\psi(0^+)\rangle = e^{-i\lambda\hat{X}/\hbar}|0\rangle$ 在 $t>0$ 时按自由谐振子演化：
\[
|\psi(t)\rangle = e^{-i\hat{H}_0 t/\hbar}|\psi(0^+)\rangle
= \sum_n c_n e^{-iE_n t/\hbar}|n\rangle.
\]
各能量本征态获得各自的相位因子 $e^{-iE_n t/\hbar}$，但展开系数 $|c_n|$ 不变。因此：
\begin{itemize}[nosep]
    \item \textbf{能量概率} $P_n = |c_n|^2$：\textbf{不随时间变化}（能量本征态测量的基本性质）
    \item \textbf{位置/动量期望值}：$\langle\hat{X}\rangle(t)$ 和 $\langle\hat{P}\rangle(t)$ 会随时间做简谐振荡（相干态的经典行为）
\end{itemize}
\end{warnbox}

\subsubsection{与位移算符的对应}

将 $\hat{X}$ 用升降算符表示：
\[
e^{-i\lambda\hat{X}/\hbar}
= \exp\!\Bigl[-\frac{i\lambda}{\hbar}x_0\frac{\hat{a}+\hat{a}^\dagger}{\sqrt{2}}\Bigr]
= \exp(\alpha\hat{a}^\dagger - \alpha^*\hat{a})
= \hat{D}(\alpha),
\]
其中
\begin{equation}
\alpha = -\frac{i\lambda x_0}{\hbar\sqrt{2}}
= -\frac{i\lambda}{\hbar}\sqrt{\frac{\hbar}{2m\omega}}.
\end{equation}

由于 $\alpha$ 是纯虚数，$\hat{D}(\alpha)$ 对应\textbf{位置空间的平移}（动量空间的 kick）。若初态为基态 $|0\rangle$，则末态为相干态 $|\alpha\rangle$。

\subsubsection{跃迁概率}

既然末态是相干态，能量测量概率直接继承泊松分布：
\begin{equation}
\boxed{P_n = \frac{e^{-\bar{n}}\,\bar{n}^{\,n}}{n!},
\qquad \bar{n} = |\alpha|^2 = \frac{\lambda^2}{2m\omega\hbar}.}
\end{equation}
特别地，基态保留概率 $P_0 = e^{-\bar{n}}$，激发总概率 $P_\text{exc}=1-e^{-\bar{n}}$，平均激发数 $\langle n\rangle = \bar{n}$。

\begin{tipbox}[瞬时脉冲 vs 其他微扰]
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\toprule
\textbf{微扰形式} & \textbf{时间演化} & \textbf{结果特征} \\
\midrule
$V\,\Theta(t)$（常微扰） & 微扰级数 & 费米黄金规则 \\
$V\,e^{-t/T}$（指数衰减） & 微扰级数 & Lorentz 线型 \\
$V\,\delta(t)$（瞬时脉冲） & \textbf{精确指数} $e^{-iV/\hbar}$ & \textbf{相干态/Poisson} \\

\end{tabular}
\end{center}
只有 $\delta(t)$ 能让 $U$ 写成闭式指数，因为积分 $\int\delta(t)\,dt$ 有限；其他形式都需微扰展开。
\end{tipbox}

\subsubsection{叠加态的时间演化}

相干态是最经典的态，但即使是简单的两能级叠加也展现出丰富的动力学。考虑
\[
|\psi(0)\rangle = \cos\theta\,|0\rangle + i\sin\theta\,|1\rangle.
\]

$t$ 时刻的态为
\[
|\psi(t)\rangle = \cos\theta\,e^{-i\omega t/2}|0\rangle + i\sin\theta\,e^{-3i\omega t/2}|1\rangle.
\]

位置与动量的期望值为
\begin{align*}
\langle\hat{X}\rangle(t)
&= \sqrt{2}\,x_0\,\text{Re}\bigl\langle\hat{a}(t)\bigr\rangle
= \sqrt{2}\,x_0\sin\theta\cos\theta\sin\omega t, \\
\langle\hat{P}\rangle(t)
&= \sqrt{2}\,p_0\,\text{Im}\bigl\langle\hat{a}(t)\bigr\rangle
= \sqrt{2}\,p_0\sin\theta\cos\theta\cos\omega t.
\end{align*}

$\langle\hat{X}\rangle$ 以 $\omega$ 振荡，振幅 $\sqrt{2}x_0\sin\theta\cos\theta$，最大值出现在 $t=(n+1/2)\pi/\omega$。

能量期望为
\[
\langle\hat{H}\rangle = \cos^2\theta\,E_0 + \sin^2\theta\,E_1
= \frac{\hbar\omega}{2}(1 + 2\sin^2\theta).
\]

动能与势能分别为
\[
\langle\hat{T}\rangle = \langle\hat{V}\rangle = \frac{\hbar\omega}{4}(1 + 2\sin^2\theta),
\]
两者相等（此特殊叠加态恰好满足 Virial 定理的等式形式，但一般叠加态不满足）。

\subsubsection{两能级叠加的概率密度振荡}

更一般的两能级叠加
\[
\psi(x,0) = A\Bigl(\psi_1(x) + \frac12\psi_2(x)\Bigr)
\]
经归一化得 $A = 2/\sqrt{5}$。时间演化给出
\[
\psi(x,t) = A\Bigl(e^{-3i\omega t/2}\psi_1(x) + \tfrac12 e^{-5i\omega t/2}\psi_2(x)\Bigr).
\]

概率密度为
\begin{equation}
|\psi(x,t)|^2 = A^2\Bigl[\psi_1^2 + \frac14\psi_2^2 + \psi_1\psi_2\cos(\omega t)\Bigr].
\end{equation}

干涉项以频率 $\omega$（即 $E_2-E_1=\hbar\omega$）振荡，导致概率密度在空间分布上周期性"流动"。这是量子相干性的直接体现：不同于经典粒子的固定轨道，量子概率在空间中来回振荡。

% 7. 带电谐振子在均匀电场中：Stark 效应与极化率
% --------------------------------------------------------
\subsection{带电谐振子在均匀电场中：Stark 效应与极化率}

一个带电荷 $q$ 的谐振子置于沿 $x$ 轴的均匀电场 $\mathcal{E}$ 中，哈密顿量为
\begin{equation}
\hat{H} = \frac{\hat{P}^2}{2m} + \frac12 m\omega^2\hat{X}^2 - q\mathcal{E}\hat{X}.
\end{equation}
其中 $-q\mathcal{E}\hat{X}$ 是电场与电偶极矩的耦合。这是一个\textbf{可精确求解}的模型，也是微扰论教学中最经典的例子之一。

\subsubsection{精确解：配方法与坐标平移}

将势能配方：
\begin{align*}
\frac12 m\omega^2\hat{X}^2 - q\mathcal{E}\hat{X}
&= \frac12 m\omega^2\Bigl(\hat{X} - \frac{q\mathcal{E}}{m\omega^2}\Bigr)^2 - \frac{q^2\mathcal{E}^2}{2m\omega^2}.
\end{align*}

引入平移坐标 $\hat{X}' = \hat{X} - q\mathcal{E}/(m\omega^2)$，注意平移不影响动量 $\hat{P}$，则
\[
\hat{H} = \frac{\hat{P}^2}{2m} + \frac12 m\omega^2\hat{X}'^2 - \frac{q^2\mathcal{E}^2}{2m\omega^2}.
\]

这正是标准谐振子哈密顿量（变量为 $\hat{X}'$）加上一个\textbf{与量子数无关的常数偏移}。因此能级和本征函数为
\begin{equation}
\boxed{E_n = \hbar\omega\Bigl(n + \frac12\Bigr) - \frac{q^2\mathcal{E}^2}{2m\omega^2},}
\qquad
\boxed{\psi_n(x) = \psi_n^{(0)}\Bigl(x - \frac{q\mathcal{E}}{m\omega^2}\Bigr),}
\end{equation}
其中 $\psi_n^{(0)}$ 是零电场时的标准谐振子波函数。

\begin{insightbox}[物理图像：电场只移动平衡位置]
与经典谐振子完全类似，均匀电场不改变振动频率，只是将平衡位置从 $x=0$ 移动到
\[
x_\text{eq} = \frac{q\mathcal{E}}{m\omega^2}.
\]
波函数形状不变，整体"跟着"平衡位置平移。所有能级的下移量相同，因此能级间距 $\hbar\omega$ 保持不变。
\end{insightbox}

\subsubsection{微扰论验证}

将 $-q\mathcal{E}\hat{X}$ 视为微扰，可以验证微扰论的各阶结果。

\textbf{一阶能量修正}：
\[
E_n^{(1)} = -q\mathcal{E}\,\langle n|\hat{X}|n\rangle = 0.
\]
这是因为谐振子本征态具有确定宇称，$\psi_n(x)$ 的奇偶性与 $n$ 相同，而被积函数 $x|\psi_n(x)|^2$ 总是奇函数，对角矩阵元恒为零。

\textbf{二阶能量修正}：利用 $\hat{X} = \frac{x_0}{\sqrt{2}}(\hat{a} + \hat{a}^\dagger)$，只有 $|n\pm 1\rangle$ 中间态有贡献：
\begin{align*}
E_n^{(2)}
&= q^2\mathcal{E}^2\sum_{m\neq n}\frac{|\langle m|\hat{X}|n\rangle|^2}{E_n - E_m} \\
&= q^2\mathcal{E}^2\left[
\frac{|\langle n-1|\hat{X}|n\rangle|^2}{\hbar\omega}
+ \frac{|\langle n+1|\hat{X}|n\rangle|^2}{-\hbar\omega}
\right] \\
&= q^2\mathcal{E}^2\cdot\frac{x_0^2}{2\hbar\omega}\bigl[n - (n+1)\bigr]
= -\frac{q^2\mathcal{E}^2}{2m\omega^2}.
\end{align*}

与精确解完全一致。这说明对谐振子，微扰级数在二阶就\textbf{收敛到精确结果}——事实上所有更高阶修正都为零，因为 $[\hat{X}, [\hat{X}, \hat{H}_0]] = \text{常数}$，微扰哈密顿量作为 $\hat{X}$ 的线性函数，通过配方法可一次性对角化。

\subsubsection{电极化率}

由能级下移量 $\Delta E = -q^2\mathcal{E}^2/(2m\omega^2)$，与极化能量的一般定义
\[
\Delta E = -\frac12 \alpha \mathcal{E}^2
\]
对比，得\textbf{静态电极化率}
\begin{equation}
\boxed{\alpha = \frac{q^2}{m\omega^2} = \frac{q^2}{k}.}
\end{equation}

另一种等价定义来自诱导偶极矩。在任意定态 $|n\rangle$ 中，位置的期望值为
\[
\langle\hat{X}\rangle_n = \int x\,|\psi_n^{(0)}(x - x_\text{eq})|^2\,dx = x_\text{eq} = \frac{q\mathcal{E}}{m\omega^2},
\]
因此诱导电偶极矩为
\[
\langle d\rangle = q\langle\hat{X}\rangle_n = \frac{q^2}{m\omega^2}\mathcal{E} = \alpha\,\mathcal{E}.
\]

\begin{formulabox}[极化率的关键结论]
\begin{itemize}[nosep]
\item 所有定态具有\textbf{相同的}极化率 $\alpha = q^2/(m\omega^2)$，与量子数 $n$ 无关。
\item 这是谐振子的特有性质。对一般势（如氢原子），极化率依赖于主量子数。
\item 经典与量子结果完全一致：质量越大、弹簧越硬（$\omega$ 越大），极化率越小。
\end{itemize}
\end{formulabox}

\subsection{半经典极限：WKB 量子化条件的验证}

谐振子还有一个 remarkable 的性质：WKB 近似对其能谱给出\textbf{精确结果}。这是二次势所特有的——WKB 波函数在转折点附近的行为与精确 Airy 函数渐近形式完全匹配。

对一维势 $V(x)=m\omega^2x^2/2$，Bohr--Sommerfeld 量子化条件（含 Maslov 修正）为
\begin{equation}
\oint p\,dx = 2\pi\hbar\Bigl(n + \frac12\Bigr).
\end{equation}

转折点由 $E=V(x)$ 定出：$x=\pm a$，$a=\sqrt{2E/m\omega^2}$。经典动量 $p(x)=\sqrt{2m(E-m\omega^2x^2/2)}$。于是
\begin{align*}
\oint p\,dx &= 2\int_{-a}^{a}\sqrt{2m\Bigl(E - \frac12m\omega^2x^2\Bigr)}\,dx \\
&= 2\sqrt{2mE}\,a\int_{-\pi/2}^{\pi/2}\cos^2\theta\,d\theta
\qquad (x = a\sin\theta) \\
&= 2\sqrt{2mE}\,\sqrt{\frac{2E}{m\omega^2}}\cdot\frac{\pi}{2}
= \frac{2\pi E}{\omega}.
\end{align*}

代入量子化条件：
\[
\frac{2\pi E}{\omega} = 2\pi\hbar\Bigl(n + \frac12\Bigr)
\quad\Longrightarrow\quad
\boxed{E_n = \hbar\omega\Bigl(n + \frac12\Bigr).}
\]

\begin{insightbox}[WKB 为什么对谐振子精确？——转折点处的相位匹配]
WKB 近似的适用条件是势在 de Broglie 波长尺度上变化缓慢。在转折点 $x=\pm a$ 处，$p\to 0$，该条件必然失效，因此需要\textbf{连接公式}（connection formula）将两侧的 WKB 解衔接起来。

\textbf{一般势的连接公式}：在转折点附近将势近似为线性 $V(x)\approx E + F(x-a)$，薛定谔方程的解是 Airy 函数。Airy 函数的渐近行为给出 WKB 相位的一个额外修正——\textbf{Maslov 修正}。对一维势，两侧各有一个转折点，每个转折点贡献 $\pi/4$ 的相位，总修正为 $\pi/2$，对应量子化条件中的 $+1/2$：
\[
\oint p\,dx = 2\pi\hbar\Bigl(n + \frac12\Bigr).
\]

\textbf{谐振子的特殊性}：谐振子的势在转折点附近\textbf{不是}线性的，而是二次的。但奇妙的是，二次势的精确解（抛物柱函数）在渐近区的相位行为，与线性近似下的 Airy 函数给出的连接结果完全相同。更深层的原因是：谐振子的作用量变量 $J = E/\omega$ 与角变量 $\theta = \omega t$ 构成的经典相空间是严格的圆，WKB 量子化本质上是将圆环周长量子化，而谐振子的量子化条件 $E_n=\hbar\omega(n+1/2)$ 恰好是这一几何结构的精确反映。对任意其他势（如 $V\propto x^4$），WKB 只能给出近似能级。
\end{insightbox}

\subsection{特殊微扰：$\hat{H}' = i\lambda[\hat{A}, \hat{H}_0]$}

一类在形式上极为简洁的微扰是
\[
\hat{H}' = i\lambda[\hat{A}, \hat{H}_0],
\qquad \lambda\in\mathbb{R},
\]
其中 $\hat{A}$ 为厄米算符。由于 $[\hat{A}, \hat{H}_0]$ 是反厄米的（$([\hat{A}, \hat{H}_0])^\dagger = -[\hat{A}, \hat{H}_0]$），乘以 $i$ 后 $\hat{H}'$ 是\textbf{厄米}的，可作为合法微扰。

\subsubsection{一阶能量修正为零}

对 $H_0$ 的本征态 $|n\rangle$：
\[
\langle n|\hat{H}'|n\rangle = i\lambda\langle n|[\hat{A}, \hat{H}_0]|n\rangle
= i\lambda(E_n - E_n)\langle n|\hat{A}|n\rangle = 0.
\]
因此\textbf{一阶能量修正恒为零}，与 $\hat{A}$ 的具体形式无关。

\subsubsection{波函数一阶修正}

微扰矩阵元
\[
\langle m|\hat{H}'|n\rangle = i\lambda(E_n - E_m)\langle m|\hat{A}|n\rangle,
\qquad m\neq n.
\]

波函数一阶修正为
\begin{align*}
|\psi_n^{(1)}\rangle
&= \sum_{m\neq n}\frac{\langle m|\hat{H}'|n\rangle}{E_n - E_m}|m\rangle \\
&= i\lambda\sum_{m\neq n}\langle m|\hat{A}|n\rangle|m\rangle \\
&= i\lambda\bigl(\hat{A} - \langle n|\hat{A}|n\rangle\bigr)|n\rangle.
\end{align*}

\begin{formulabox}[简洁结果]
\[
\boxed{|\psi_n^{(1)}\rangle = i\lambda\bigl(\hat{A} - \langle\hat{A}\rangle_n\bigr)|n\rangle}
\]
一阶修正正比于 $\hat{A}$ 在 $|n\rangle$ 上的「涨落」，不需要求和。
\end{formulabox}

\subsubsection{二阶能量修正}

\begin{align*}
E_n^{(2)}
&= \sum_{m\neq n}\frac{|\langle m|\hat{H}'|n\rangle|^2}{E_n - E_m} \\
&= -\lambda^2\sum_{m\neq n}(E_m - E_n)|\langle m|\hat{A}|n\rangle|^2.
\end{align*}

对谐振子取 $\hat{A}=\hat{X}$，利用选择定则 $|\langle m|\hat{X}|n\rangle|^2 = (x_0^2/2)\bigl[(n+1)\delta_{m,n+1} + n\delta_{m,n-1}\bigr]$：
\[
E_n^{(2)} = -\lambda^2\cdot\frac{x_0^2}{2}\bigl[\hbar\omega(n+1) - \hbar\omega n\bigr]
= -\frac{\lambda^2\hbar^2}{2m}.
\]

\begin{insightbox}[与 TRK 的联系]
二阶修正中的求和 $\sum_{m\neq n}(E_m-E_n)|\langle m|\hat{A}|n\rangle|^2$ 正是 TRK 求和规则的推广形式。对任意 $\hat{A}$ 满足 $[[\hat{H},\hat{A}],\hat{A}]\propto I$，该求和为常数，导致二阶修正与量子数 $n$ 无关。这是此类微扰的一个显著特征。
\end{insightbox}

% 8. 求和规则：矩阵元的系统约束

\subsection{求和规则：矩阵元的系统约束}

能量表象中的矩阵元并非彼此独立，它们受到基本对易关系的约束。定义\textbf{振子强度}
\begin{equation}
f_n = \frac{2m}{\hbar^2}(E_n - E_0)\,\bigl|\langle n|\hat{X}|0\rangle\bigr|^2,
\end{equation}
可以证明对任意一维束缚势都有
\begin{equation}
\boxed{\sum_{n=0}^{\infty} f_n = 1.}
\end{equation}
这就是 Thomas--Reiche--Kuhn (TRK) 求和规则。

\subsubsection{普适证明：双重对易子法}

TRK 求和规则对任意一维束缚势均成立，最简洁的证法是利用双重对易子 $[[\hat{H},\hat{X}],\hat{X}]$。

\begin{derivationbox}[双重对易子法]
首先计算对易子：
\[
[\hat{H}, \hat{X}] = \Bigl[\frac{\hat{P}^2}{2m}, \hat{X}\Bigr] = \frac{i\hbar}{m}\hat{P}.
\]
再对易一次：
\[
[[\hat{H}, \hat{X}], \hat{X}] = \Bigl[\frac{i\hbar}{m}\hat{P}, \hat{X}\Bigr] = \frac{i\hbar}{m}(i\hbar) = \frac{\hbar^2}{m}.
\]

另一方面，对任意算符有恒等式
\[
[[\hat{H},\hat{X}],\hat{X}] = \hat{H}\hat{X}^2 - 2\hat{X}\hat{H}\hat{X} + \hat{X}^2\hat{H}.
\]
取基态期望值并插入完备基 $\sum_n|n\rangle\langle n|=I$：
\begin{align*}
\frac{\hbar^2}{m}
&= \sum_n\Bigl(E_0|\langle n|\hat{X}|0\rangle|^2 - 2E_n|\langle n|\hat{X}|0\rangle|^2 + E_0|\langle n|\hat{X}|0\rangle|^2\Bigr) \\
&= 2\sum_n(E_0 - E_n)\bigl|\langle n|\hat{X}|0\rangle\bigr|^2.
\end{align*}
整理得
\[
\sum_n(E_n - E_0)\bigl|\langle n|\hat{X}|0\rangle\bigr|^2 = \frac{\hbar^2}{2m},
\]
乘以 $2m/\hbar^2$ 即得 $\sum_n f_n = 1$。
\end{derivationbox}

\subsubsection{在谐振子中验证}

对谐振子，上述普适关系有特别简洁的验证。基态动能期望为
\[
\langle 0|\hat{T}|0\rangle = \frac{E_0}{2} = \frac{\hbar\omega}{4}.
\]

基态位置方差为
\[
(\Delta X)_0^2 = \langle 0|\hat{X}^2|0\rangle - \langle 0|\hat{X}|0\rangle^2 = \frac{x_0^2}{2} - 0 = \frac{\hbar}{2m\omega}.
\]

由 Virial 定理 $\langle\hat{T}\rangle = \langle\hat{V}\rangle$，有
\[
\langle 0|\hat{T}|0\rangle = \frac12 m\omega^2 (\Delta X)_0^2
= \frac12 m\omega^2 \cdot \frac{\hbar}{2m\omega} = \frac{\hbar\omega}{4}.\quad\checkmark
\]

第一激发态与基态的能量差为 $E_1-E_0=\hbar\omega$。利用 TRK 求和规则的左侧（只有 $n=1$ 有贡献）：
\[
(E_1-E_0)\bigl|\langle 1|\hat{X}|0\rangle\bigr|^2 = \hbar\omega \cdot \frac{x_0^2}{2}
= \hbar\omega \cdot \frac{\hbar}{2m\omega} = \frac{\hbar^2}{2m}.\quad\checkmark
\]

这正是双重对易子给出的普适值 $\hbar^2/(2m)$，与势场形式无关。

此外，谐振子的选择定则 $\Delta n=\pm 1$ 意味着求和规则中只有 $n=1$ 一项有贡献：
\[
f_1 = \frac{2m}{\hbar^2}\cdot\hbar\omega\cdot\frac{x_0^2}{2}
= \frac{2m}{\hbar^2}\cdot\hbar\omega\cdot\frac{\hbar}{2m\omega} = 1,
\]
而 $f_{n\neq 1}=0$，故 $\sum_n f_n = 1$。谐振子"恰好"由第一激发态贡献全部强度。

\subsubsection{推广：任意初态 $|m\rangle$}

上述证明可推广到以任意本征态 $|m\rangle$ 为初态的情形。利用 $[[\hat{H},\hat{X}],\hat{X}]=\hbar^2/m$ 并在 $|m\rangle$ 中取期望：
\[
\langle m|[[\hat{H},\hat{X}],\hat{X}]|m\rangle = \frac{\hbar^2}{m}.
\]

展开左边并插入完备基：
\begin{align*}
\langle m|[[\hat{H},\hat{X}],\hat{X}]|m\rangle
&= \sum_n\langle m|(\hat{H}\hat{X}-\hat{X}\hat{H})|n\rangle\langle n|\hat{X}|m\rangle
- \sum_n\langle m|\hat{X}|n\rangle\langle n|(\hat{H}\hat{X}-\hat{X}\hat{H})|m\rangle \\
&= \sum_n(E_m-E_n)|\langle n|\hat{X}|m\rangle|^2
- \sum_n(E_n-E_m)|\langle m|\hat{X}|n\rangle|^2 \\
&= 2\sum_n(E_m-E_n)|\langle n|\hat{X}|m\rangle|^2.
\end{align*}

注意 $\langle n|\hat{X}|m\rangle$ 为实矩阵元（实波函数），$|\langle n|\hat{X}|m\rangle|^2=|\langle m|\hat{X}|n\rangle|^2$。整理得
\begin{equation}
\boxed{\sum_n(E_n-E_m)\bigl|\langle n|\hat{X}|m\rangle\bigr|^2 = \frac{\hbar^2}{2m}.}
\end{equation}

对基态 $m=0$，回到 TRK 求和规则。对任意 $m$，该式表明：从 $|m\rangle$ 出发，到所有高能级的"加权跃迁强度"之和是普适常数，与 $m$ 无关。

\begin{formulabox}[基态动能的下界]
由不确定性原理 $\Delta P\,\Delta X\geq\hbar/2$ 及 $\langle T\rangle_0=\langle P^2\rangle_0/2m\geq(\Delta P)_0^2/2m$，得
\[
\langle T\rangle_0 \geq \frac{\hbar^2}{8m(\Delta X)_0^2}.
\]

结合上式取 $m=0$ 时 $\sum_{n>0}(E_n-E_0)|\langle n|\hat{X}|0\rangle|^2=\hbar^2/(2m)$，并利用 $(E_1-E_0)\sum_{n>0}|\langle n|\hat{X}|0\rangle|^2\leq\sum_{n>0}(E_n-E_0)|\langle n|\hat{X}|0\rangle|^2$。其中
\[
\sum_{n>0}|\langle n|\hat{X}|0\rangle|^2
= \langle 0|\hat{X}^2|0\rangle - |\langle 0|\hat{X}|0\rangle|^2
= (\Delta X)_0^2
\]
（最后一步对对称势 $V(x)=V(-x)$ 有 $x_{00}=\langle 0|\hat{X}|0\rangle=0$，故 $|\langle 0|\hat{X}|0\rangle|^2=0$，而 $\langle 0|\hat{X}^2|0\rangle=\sum_n|\langle n|\hat{X}|0\rangle|^2$ 来自插入完备性关系 $\sum_n|n\rangle\langle n|=\hat{1}$），可得
\[
(E_1-E_0)(\Delta X)_0^2 \leq \frac{\hbar^2}{2m}
\quad\Longrightarrow\quad
\langle T\rangle_0 \geq \frac{E_1-E_0}{4}.
\]

对谐振子：$\langle T\rangle_0=\hbar\omega/4=(E_1-E_0)/4$，等号成立。对无限深势阱，不等式严格成立。
\end{formulabox}

% 9. 对称性工具：HF 定理与 Virial 定理
% --------------------------------------------------------
\subsection{对称性工具：HF 定理与 Virial 定理}

谐振子的能谱和期望值还可通过参数依赖性和尺度对称性导出，无需解微分方程。

\subsubsection{Hellmann--Feynman 定理}

若 $H(\lambda)$ 含参，$|n(\lambda)\rangle$ 归一化，则
\begin{equation}
\boxed{\frac{\partial E_n}{\partial\lambda}
= \Big\langle n\Big|\frac{\partial H}{\partial\lambda}\Big|n\Big\rangle.}
\end{equation}

将频率 $\omega$ 视为参数。谐振子 $H = \hat{P}^2/2m + m\omega^2\hat{X}^2/2$，故
\[
\frac{\partial H}{\partial\omega} = m\omega\hat{X}^2.
\]
HF 定理给出
\[
\frac{\partial E_n}{\partial\omega}
= m\omega\langle n|\hat{X}^2|n\rangle
= m\omega\cdot\frac{(n+\tfrac12)\hbar}{m\omega}
= \Bigl(n+\frac12\Bigr)\hbar.
\]
积分即得 $E_n = \hbar\omega(n+1/2)$——不解薛定谔方程，仅从参数依赖性就恢复了能谱。

\subsubsection{Virial 定理}

Virial 定理可以从 HF 定理自然导出，也可以通过直接对易子证明。这里给出最直观的推导——\textbf{尺度变换法}。

\begin{derivationbox}[尺度变换法推导 Virial 定理]
令坐标作标度变换 $x\to\lambda x$，则动量 $p = -i\hbar\,d/dx$ 变为 $p\to\lambda^{-1}p$（因为 $d/d(\lambda x) = \lambda^{-1}d/dx$）。含参哈密顿量
\[
H(\lambda) = \lambda^{-2}\frac{P^2}{2m} + V(\lambda x).
\]

对 $\lambda$ 求导：
\[
\frac{\partial H}{\partial\lambda} = -2\lambda^{-3}\hat{T} + x\,V'(\lambda x).
\]

在物理点 $\lambda=1$ 处，能量对尺度变换不变：$\partial E/\partial\lambda|_{\lambda=1}=0$（因为 $\lambda=1$ 就是真实系统）。由 HF 定理
\[
0 = \Big\langle -2\hat{T} + x\frac{dV}{dx}\Big\rangle
\quad\Longrightarrow\quad
\boxed{2\langle\hat{T}\rangle = \Big\langle x\frac{dV}{dx}\Big\rangle.}
\]

对幂次势 $V\propto x^s$，有 $xV'(x) = sV(x)$，故 $2\langle\hat{T}\rangle = s\langle V\rangle$。
\end{derivationbox}

对谐振子 $V = m\omega^2x^2/2$，幂次 $s=2$，因此
\[
\langle\hat{T}\rangle = \langle V\rangle = \frac{E_n}{2}.
\]

由此直接读出坐标与动量的方差：
\begin{align}
\Big\langle\frac12m\omega^2\hat{X}^2\Big\rangle &= \frac{E_n}{2}
\quad\Longrightarrow\quad
\langle\hat{X}^2\rangle = \frac{(n+\tfrac12)\hbar}{m\omega}, \\
\Big\langle\frac{\hat{P}^2}{2m}\Big\rangle &= \frac{E_n}{2}
\quad\Longrightarrow\quad
\langle\hat{P}^2\rangle = (n+\tfrac12)m\omega\hbar.
\end{align}

\begin{formulabox}[Virial 定理在谐振子上的结果]
\[
\boxed{\Big\langle\frac{\hat{P}^2}{2m}\Big\rangle = \Big\langle\frac12m\omega^2\hat{X}^2\Big\rangle
= \frac{\hbar\omega}{2}\Bigl(n+\frac12\Bigr) = \frac{E_n}{2}}
\]
\textbf{与矩阵元法完全一致}，但 Virial 的优势在于：一旦知道总能量 $E_n$，立即拆分为动能与势能，\textbf{无需计算任何矩阵元}。
\end{formulabox}

\begin{insightbox}[HF 与 Virial 的互补性]
HF 定理回答"能量如何随参数变化"，Virial 定理回答"能量如何在动能与势能间分配"。两者都只需要哈密顿量的结构信息，不依赖波函数的具体形式。对谐振子这类简单模型，它们是快速验证结果的自洽工具；对复杂模型（如氢原子），它们是从能级公式提取期望值的利器。
\end{insightbox}

% 10. 多体推广：谐振子阱中的全同费米子
% --------------------------------------------------------
\subsection{多体推广：谐振子阱中的全同费米子}

将单粒子谐振子模型推广到 $N$ 个无自旋全同费米子，两体相互作用为排斥 $\delta$ 势：
\[
\hat{V} = \frac{\lambda}{2}\sum_{i\neq j}\delta(x_i-x_j),
\qquad \lambda>0.
\]

\subsubsection{零级近似：Slater 行列式与能级填充}

单粒子能级 $\varepsilon_n = \hbar\omega(n+1/2)$。由于 Pauli 不相容，$N$ 个费米子必须占据不同的单粒子态。基态将最低 $N$ 个能级 $0,1,\dots,N-1$ 各填一个粒子。总能量为
\begin{align}
E_0^{(0)} &= \hbar\omega\sum_{n=0}^{N-1}\Bigl(n+\frac12\Bigr)
= \hbar\omega\Bigl(\frac{(N-1)N}{2} + \frac{N}{2}\Bigr) \\
&= \hbar\omega\cdot\frac{N^2}{2}
= \boxed{\frac{\hbar\omega N^2}{2}.}
\end{align}

零级波函数是 Slater 行列式
\[
\Psi_0(x_1,\dots,x_N) = \frac{1}{\sqrt{N!}}\det\bigl[\psi_i(x_j)\bigr],
\qquad i,j=0,1,\dots,N-1.
\]
行列式的反对称性保证：交换任意两粒子坐标 $x_i\leftrightarrow x_j$ 时，$\Psi$ 变号。因此当 $x_i=x_j$ 时，两行相同，行列式为零：
\begin{equation}
\Psi(x_i=x_j) = 0.
\end{equation}

\subsubsection{$\delta$ 相互作用的一级修正}

$\delta$ 势的矩阵元正比于 $|\Psi|^2$ 在接触点的值。但费米子波函数在两粒子坐标重合处严格为零，故
\[
\langle\Psi|\delta(x_i-x_j)|\Psi\rangle = \int|\Psi(\dots,x,\dots,x,\dots)|^2\,dx = 0.
\]

$\delta$ 相互作用对费米子的一级修正\textbf{严格为零}。这是 Pauli 不相容原理的直接后果——费米子"感觉不到"短程接触相互作用。

\begin{insightbox}[为什么费米子感受不到 $\delta$ 排斥？]
从波函数角度：费米子波函数的节点结构强制在两粒子接触处出现零点。

从物理角度：Pauli 原理已经在费米子间建立了"有效排斥"，使它们彼此回避。额外的短程 $\delta$ 排斥试图让粒子更远离彼此，但粒子已经处于尽可能远离的状态（每个占据不同量子数），因此没有进一步修正的余地。
\end{insightbox}

\subsubsection{激发态与 Virial 定理}

\textbf{第一激发态}：将最高占据粒子从 $N-1$ 激发到 $N$，占据 $0,1,\dots,N-2,N$。能量
\[
E_1 = \hbar\omega\Bigl(\sum_{n=0}^{N-2}n + N + \frac{N}{2}\Bigr)
= \hbar\omega\cdot\frac{N^2+2}{2}.
\]
简并度为 1（只有一种移动方式）。

\textbf{第二激发态}：总激励子数为 2，有两种等价组态：
\begin{itemize}[nosep]
    \item 将 $N-1$ 激发到 $N+1$：占据 $0,1,\dots,N-2,N+1$
    \item 将 $N-2,N-1$ 分别激发到 $N-1,N$：占据 $0,1,\dots,N-3,N-1,N$
\end{itemize}
两种组态能量均为
\[
E_2 = \hbar\omega\cdot\frac{N^2+4}{2},
\]
简并度为 2。

对二次势，Virial 定理给出 $\langle V_\text{ho}\rangle = E/2$（在 $\delta$ 修正为零的近似下）。由 $V_\text{ho} = \frac12m\omega^2\sum_i x_i^2$ 得
\begin{equation}
\boxed{\Big\langle\sum_{i=1}^N x_i^2\Big\rangle = \frac{E}{m\omega^2}
= \frac{\hbar}{2m\omega}(N^2 + 2k),}
\end{equation}
其中 $k=0,2,4$ 分别对应基态、第一激发态、第二激发态。

\begin{insightbox}[费米压与空间扩展]
$\langle\sum x_i^2\rangle \propto N^2$ 体现了 Pauli 排斥导致的"费米压"：为了占据不同的量子态，费米子必须分布在越来越大的空间区域内。相比之下，$N$ 个玻色子凝聚在基态时，$\langle\sum x_i^2\rangle \propto N$（单粒子尺度）。这种 $N^2$ 与 $N$ 的差异是费米统计最直观的宏观表现之一。
\end{insightbox}

% 11. 一维特殊性：束缚态的非简并性

\subsection{一维特殊性：束缚态的非简并性}

谐振子每个能级恰好只有一个量子态（除自旋外），这是一维束缚态普遍定理的特例。

\subsubsection{Wronskian 论证}

考虑任意一维实势 $V(x)$ 的定态方程。设 $\psi(x)$ 与 $\phi(x)$ 同为能量 $E$ 的解，则
\[
\psi'' = -\frac{2m}{\hbar^2}(E-V)\psi,
\qquad
\phi'' = -\frac{2m}{\hbar^2}(E-V)\phi.
\]

两式交叉相乘相减得
\[
\psi''\phi - \phi''\psi = 0
\quad\Longrightarrow\quad
\frac{d}{dx}\bigl(\psi'\phi - \phi'\psi\bigr) = 0.
\]
因此 Wronskian $W = \psi'\phi - \phi'\psi = \text{const}$。对束缚态，$x\to\pm\infty$ 时 $\psi,\phi\to 0$，故 $W\to 0$，从而 $W\equiv 0$。

$W=0$ 意味着 $\phi'\psi = \psi'\phi$，即 $(\ln\phi/\psi)'=0$，积分得 $\phi = c\psi$。两个解线性相关，能量 $E$ 的束缚态\textbf{非简并}。

\subsubsection{实波函数与宇称}

若 $V(x)$ 为实函数，薛定谔方程系数为实，则 $\psi^*(x)$ 也是同能量解。由非简并性，$\psi^* = e^{i\delta}\psi$。再取复共轭得 $|e^{i\delta}|^2=1$。通过调整整体相位 $e^{-i\delta/2}$，总可使波函数变为实函数。

\subsubsection{宇称定理}

若势能满足 $V(-x)=V(x)$（空间反演对称），则哈密顿量与宇称算符 $\hat{\Pi}$ 对易：$[\hat{H},\hat{\Pi}]=0$。因此能量本征态可取为宇称算符的共同本征态。

设 $\psi(x)$ 为能量 $E$ 的本征解，则 $\psi(-x)$ 也是同能量解。由非简并性，$\psi(-x)=c\psi(x)$。再反演一次：$\psi(x)=c^2\psi(x)$，故 $c^2=1$，即 $c=\pm 1$。

因此一维对称势的束缚态必具有确定宇称：
\begin{itemize}[nosep]
    \item $c=+1$：偶宇称 $\psi(-x)=\psi(x)$（如谐振子 $n=0,2,4,\dots$）
    \item $c=-1$：奇宇称 $\psi(-x)=-\psi(x)$（如谐振子 $n=1,3,5,\dots$）
\end{itemize}

\begin{formulabox}[一维束缚态三大定理]
\begin{enumerate}
    \item \textbf{非简并定理}：一维束缚态能量非简并
    \item \textbf{实波函数定理}：可取为实函数（至多差一个全局相因子）
    \item \textbf{宇称定理}：若 $V(x)=V(-x)$，则本征态具有确定宇称 $(-1)^n$
\end{enumerate}
谐振子 $H_n(\xi)$ 同时满足三条：实函数、确定宇称 $(-1)^n$、无简并。
\end{formulabox}

\begin{insightbox}[一维 vs 高维]
一维束缚态非简并的根源是边界条件过于"严格"——两端趋于零的要求不允许两个线性无关解同时满足。在三维中心势中，径向方程只要求 $r\to\infty$ 时衰减，而原点处正则性条件较松，因此同一能量可以容纳不同角动量 $l$ 的态（如氢原子的 $2s$ 与 $2p$）。谐振子在三维中同样存在简并：各向同性时 $E = \hbar\omega(2n_r+l+3/2)$，不同 $(n_r,l)$ 组合可对应同一能量。
\end{insightbox}

\subsection{压缩算符与标度变换}

压缩算符是谐振子代数自然延伸出的另一个核心结构。定义
\begin{equation}
\hat{S}(\xi) = \exp\!\Bigl\{\frac12\bigl(\xi^*\hat{a}^2 - \xi(\hat{a}^\dagger)^2\bigr)\Bigr\},
\qquad \xi\in\mathbb{C}.
\end{equation}
当 $\xi=r$ 为实数时，$\hat{S}(r)$ 实现对坐标和动量的\textbf{标度变换}（squeezing）：一个方向压缩、正交方向膨胀，保持最小不确定关系。

\subsubsection{标度变换的推导}

令 $\hat{a}(r) = \hat{S}^\dagger(r)\hat{a}\hat{S}(r)$。对参数 $r$ 求导，利用
\[
\frac{d\hat{a}(r)}{dr} = [\hat{a}(r), \hat{G}],
\qquad \hat{G} = \frac12\bigl(\hat{a}^2 - (\hat{a}^\dagger)^2\bigr),
\]
计算对易子：
\[
[\hat{a}, \hat{G}] = \frac12[\hat{a}, \hat{a}^2 - (\hat{a}^\dagger)^2] = -\hat{a}^\dagger,
\qquad
[\hat{a}^\dagger, \hat{G}] = -\hat{a}.
\]

得到耦合方程组：
\[
\frac{d\hat{a}}{dr} = -\hat{a}^\dagger(r),
\qquad
\frac{d\hat{a}^\dagger}{dr} = -\hat{a}(r).
\]

再求导一次得 $d^2\hat{a}/dr^2 = \hat{a}$，通解为双曲函数：
\begin{equation}
\boxed{\hat{a}(r) = \hat{a}\cosh r - \hat{a}^\dagger\sinh r,
\qquad
\hat{a}^\dagger(r) = \hat{a}^\dagger\cosh r - \hat{a}\sinh r.}
\end{equation}

\subsubsection{对坐标与动量的作用}

将 $\hat{X} = x_0(\hat{a}+\hat{a}^\dagger)/\sqrt{2}$ 和 $\hat{P} = p_0(\hat{a}-\hat{a}^\dagger)/(i\sqrt{2})$ 代入：
\begin{align*}
\hat{S}^\dagger(r)\hat{X}\hat{S}(r)
&= \frac{x_0}{\sqrt{2}}\bigl[\hat{a}(r)+\hat{a}^\dagger(r)\bigr] \\
&= \frac{x_0}{\sqrt{2}}(\hat{a}+\hat{a}^\dagger)(\cosh r - \sinh r)
= e^{-r}\hat{X}, \\
\hat{S}^\dagger(r)\hat{P}\hat{S}(r)
&= \frac{p_0}{i\sqrt{2}}\bigl[\hat{a}(r)-\hat{a}^\dagger(r)\bigr] \\
&= \frac{p_0}{i\sqrt{2}}(\hat{a}-\hat{a}^\dagger)(\cosh r + \sinh r)
= e^{r}\hat{P}.
\end{align*}

\begin{formulabox}[压缩算符的核心结论]
\[
\boxed{\hat{S}^\dagger(r)\hat{X}\hat{S}(r) = e^{-r}\hat{X},
\qquad
\hat{S}^\dagger(r)\hat{P}\hat{S}(r) = e^{r}\hat{P}}
\]
比例系数 $e^{-r}$ 与 $e^{r}$ 互为倒数，乘积为 $1$。这保证了不确定关系
\[
\Delta X\,\Delta P = \frac{\hbar}{2}
\]
仍然成立（虽然 $\Delta X$ 被压缩、$\Delta P$ 被膨胀）。
\end{formulabox}

\subsubsection{物理图像与 BdG 联系}

\begin{insightbox}[相空间中的"揉面团"]
真空态在相空间中近似为一个圆（最小不确定圆）。压缩算符将其挤压成椭圆：$X$ 方向压缩 $e^{-r}$，$P$ 方向拉伸 $e^{r}$，但面积保持不变（辛结构守恒）。

当 $\xi=re^{i\theta}$ 为复数时，椭圆的长轴不再与 $X,P$ 轴对齐，而是旋转了 $\theta/2$ 角度。此时
\[
\hat{S}^\dagger(\xi)\hat{a}\hat{S}(\xi)
= \hat{a}\cosh r - \hat{a}^\dagger e^{i\theta}\sinh r,
\]
标度变换沿倾斜轴进行。
\end{insightbox}

\subsubsection{Bogoliubov 对角化的详细推导}

压缩变换与 Bogoliubov 变换的数学结构完全平行。考虑玻色子配对哈密顿量（吸引相互作用，注意配对项的负号）：
\[
\hat{H} = \omega\hat{a}^\dagger\hat{a} - \frac{\Delta}{2}\bigl(\hat{a}^2 + (\hat{a}^\dagger)^2\bigr),
\qquad \Delta>0.
\]

引入幺正变换 $\hat{b} = \hat{S}^\dagger(r)\hat{a}\hat{S}(r) = \hat{a}\cosh r - \hat{a}^\dagger\sinh r$，其逆变换为
\[
\hat{a} = \hat{b}\cosh r + \hat{b}^\dagger\sinh r,
\qquad
\hat{a}^\dagger = \hat{b}^\dagger\cosh r + \hat{b}\sinh r.
\]

代入哈密顿量。先算 $\hat{a}^\dagger\hat{a}$：
\begin{align*}
\hat{a}^\dagger\hat{a}
&= (\hat{b}^\dagger\cosh r + \hat{b}\sinh r)(\hat{b}\cosh r + \hat{b}^\dagger\sinh r) \\
&= \hat{b}^\dagger\hat{b}\cosh^2 r + (\hat{b}^\dagger)^2\sinh r\cosh r + \hat{b}^2\sinh r\cosh r + \hat{b}\hat{b}^\dagger\sinh^2 r \\
&= \hat{b}^\dagger\hat{b}\cosh(2r) + \frac{\hat{b}^2+(\hat{b}^\dagger)^2}{2}\sinh(2r) + \sinh^2 r.
\end{align*}

再算 $\hat{a}^2+(\hat{a}^\dagger)^2$：
\begin{align*}
\hat{a}^2+(\hat{a}^\dagger)^2
&= (\hat{b}^2+(\hat{b}^\dagger)^2)(\cosh^2 r+\sinh^2 r) + 2(2\hat{b}^\dagger\hat{b}+1)\sinh r\cosh r \\
&= (\hat{b}^2+(\hat{b}^\dagger)^2)\cosh(2r) + (2\hat{b}^\dagger\hat{b}+1)\sinh(2r).
\end{align*}

合并得
\begin{align*}
\hat{H}'
&= \omega\Bigl[\hat{b}^\dagger\hat{b}\cosh(2r) + \frac{\hat{b}^2+(\hat{b}^\dagger)^2}{2}\sinh(2r) + \sinh^2 r\Bigr] \\
&\quad - \frac{\Delta}{2}\Bigl[(\hat{b}^2+(\hat{b}^\dagger)^2)\cosh(2r) + (2\hat{b}^\dagger\hat{b}+1)\sinh(2r)\Bigr].
\end{align*}

要求非对角项 $(\hat{b}^2+(\hat{b}^\dagger)^2)$ 的系数为零：
\[
\frac{\omega}{2}\sinh(2r) - \frac{\Delta}{2}\cosh(2r) = 0
\quad\Longrightarrow\quad
\boxed{\tanh(2r) = \frac{\Delta}{\omega}.}
\]

此时 $b^\dagger b$ 的系数为
\[
\omega\cosh(2r) - \Delta\sinh(2r)
= \omega\cdot\frac{\omega}{E} - \Delta\cdot\frac{\Delta}{E}
= \frac{\omega^2-\Delta^2}{E} = E.
\]

常数项来源于 $a^\dagger a$ 中的 $\sinh^2 r$ 和 $a^2+(a^\dagger)^2$ 中的 $\sinh(2r)$：
\begin{align*}
C &= \omega\sinh^2 r - \frac{\Delta}{2}\sinh(2r) \\
&= \omega\cdot\frac{\cosh(2r)-1}{2} - \frac{\Delta}{2}\sinh(2r) \\
&= \frac{1}{2}\bigl(\omega\cosh(2r) - \Delta\sinh(2r)\bigr) - \frac{\omega}{2} \\
&= \frac{E-\omega}{2}.
\end{align*}

因此对角化后的哈密顿量为
\begin{equation}
\boxed{\hat{H}' = E\,\hat{b}^\dagger\hat{b} + \frac{E-\omega}{2},
\qquad E = \sqrt{\omega^2-\Delta^2}.}
\end{equation}

\begin{insightbox}[玻色子 vs 费米子 BdG 变换的数学根源]
\begin{itemize}[nosep]
    \item \textbf{玻色子}：$[\hat{b},\hat{b}^\dagger]=1$。设 $\hat{b}=u\hat{a}+v\hat{a}^\dagger$，由 $[\hat{b},\hat{b}^\dagger]=|u|^2-|v|^2=1$ 得 $u^2-v^2=1$，用\textbf{双曲函数} $(\cosh r, \sinh r)$ 参数化。
    \item \textbf{费米子}：$\{\hat{c},\hat{c}^\dagger\}=1$。设 $\hat{c}=u\hat{\gamma}+v\hat{\gamma}^\dagger$，由 $\{\hat{c},\hat{c}^\dagger\}=|u|^2+|v|^2=1$ 得 $u^2+v^2=1$，用\textbf{三角函数} $(\cos\theta, \sin\theta)$ 参数化。
\end{itemize}
两种变换在数学结构上完全平行，都是"旋转"产生/湮灭算符以消除哈密顿量中的非对角配对项。区别仅在于：玻色子的"旋转"是双曲型的（保差 $u^2-v^2=1$，类似洛伦兹变换），而费米子是圆型的（保和 $u^2+v^2=1$，类似普通转动）。这正是玻色统计与费米统计在算符代数层面的深刻体现。
\end{insightbox}


\subsubsection{【Exam】自旋-谐振子耦合的 Schrieffer-Wolff 变换}

考虑如下哈密顿量
\begin{equation}
\hat{H} = \hat{H}_0 + \hat{V},
\qquad
\hat{H}_0 = \hbar\nu\,\hat{a}^\dagger\hat{a} + \frac12\hbar\omega_\sigma\,\hat{\sigma}_z,
\qquad
\hat{V} = \frac12\eta\hbar\nu\omega\,(\hat{a}+\hat{a}^\dagger)\hat{\sigma}_z,
\end{equation}
其中 $\nu<\omega$，$\eta$ 为小量。这是\textbf{量子光学中光与两能级原子耦合}的简化模型：谐振子模描述光子场，$\hat{\sigma}_z$ 描述原子的能级劈裂。

\begin{derivationbox}[三步求解]
\textbf{（1）二阶微扰论求能级}

未微扰本征态 $|n,\sigma\rangle$，$\sigma=\pm 1$ 对应 $\sigma_z$ 本征值。未微扰能量
\[
E_{n,\sigma}^{(0)} = \hbar\nu n + \frac12\hbar\omega_\sigma\,\sigma.
\]

由于 $(\hat{a}+\hat{a}^\dagger)$ 只连接 $n\leftrightarrow n\pm 1$，对角元为零，故一阶修正 $E_{n,\sigma}^{(1)}=0$。

二阶修正只涉及 $|n\pm 1,\sigma\rangle$ 中间态：
\begin{align*}
|\langle n-1,\sigma|\hat{V}|n,\sigma\rangle|^2 &= \frac14\eta^2\hbar^2\nu^2\omega^2\,n, \\
|\langle n+1,\sigma|\hat{V}|n,\sigma\rangle|^2 &= \frac14\eta^2\hbar^2\nu^2\omega^2\,(n+1).
\end{align*}
能量分母分别为 $\hbar\nu$ 和 $-\hbar\nu$，故
\begin{equation}
\boxed{E_{n,\sigma}^{(2)} = \frac14\eta^2\hbar^2\nu^2\omega^2\Bigl(\frac{n}{\hbar\nu} - \frac{n+1}{\hbar\nu}\Bigr) = -\frac14\eta^2\hbar\nu\,\omega^2.}
\end{equation}
\textbf{Remarkable}：$n$ 的依赖完全抵消，所有能级的二阶下移量相同。

\textbf{（2）构造消除微扰的幺正变换}

引入反厄米算符
\[
\hat{S} = \frac12\frac{\eta}{\omega}\,(\hat{a}^\dagger-\hat{a})\hat{\sigma}_z,
\qquad \hat{S}^\dagger = -\hat{S}.
\]
计算 $[\hat{S},\hat{H}_0]$。由于 $\hat{\sigma}_z$ 与玻色算符对易，且
\[
[(\hat{a}^\dagger-\hat{a}),\hat{a}^\dagger\hat{a}] = -(\hat{a}+\hat{a}^\dagger),
\]
得
\begin{equation}
\boxed{[\hat{S},\hat{H}_0] = \frac12\frac{\eta}{\omega}\cdot\hbar\nu\cdot\bigl[-(\hat{a}+\hat{a}^\dagger)\bigr]\hat{\sigma}_z = -\hat{V}.}
\end{equation}
\textbf{关键发现}：$\hat{S}$ 与 $\hat{H}_0$ 的对易子恰好抵消微扰 $\hat{V}$（差一个负号）。这是 Schrieffer-Wolff 变换的核心思想——选取 $\hat{S}$ 使得 $[\hat{S},\hat{H}_0] = -\hat{V}$，从而在 BCH 展开中一阶消除微扰。

\textbf{（3）BCH 展开到 $\eta^2$}

利用
\[
e^{\hat{S}}\hat{H}e^{-\hat{S}} = \hat{H} + [\hat{S},\hat{H}] + \frac12[\hat{S},[\hat{S},\hat{H}]] + \cdots,
\]
已得 $[\hat{S},\hat{H}_0]=-\hat{V}$。再算
\begin{align*}
[\hat{S},\hat{V}]
&= \Bigl(\frac12\frac{\eta}{\omega}\Bigr)\Bigl(\frac12\eta\hbar\nu\omega\Bigr)
[(\hat{a}^\dagger-\hat{a})\hat{\sigma}_z,\,(\hat{a}+\hat{a}^\dagger)\hat{\sigma}_z] \\
&= \frac14\eta^2\hbar\nu\,[(\hat{a}^\dagger-\hat{a}),(\hat{a}+\hat{a}^\dagger)]\,\hat{\sigma}_z^2 \\
&= \frac14\eta^2\hbar\nu\cdot(-2)\cdot 1 = -\frac12\eta^2\hbar\nu.
\end{align*}
（用了 $[\hat{a}^\dagger,\hat{a}]=-1$，$\hat{\sigma}_z^2=1$）。注意原题中 $\hat{V}$ 含额外因子 $\omega$，因此 $[\hat{S},\hat{V}] = -\frac12\eta^2\hbar\nu\,\omega^2$。这是一个 \textbf{c 数}，因此更高阶对易子 $[\hat{S},[\hat{S},\hat{V}]]=0$。

于是
\begin{align*}
[\hat{S},\hat{H}] &= [\hat{S},\hat{H}_0] + [\hat{S},\hat{V}] = -\hat{V} - \frac12\eta^2\hbar\nu\,\omega^2, \\
\frac12[\hat{S},[\hat{S},\hat{H}]] &= \frac12\bigl(-[\hat{S},\hat{V}]\bigr) = \frac14\eta^2\hbar\nu\,\omega^2.
\end{align*}

合并：
\begin{align*}
e^{\hat{S}}\hat{H}e^{-\hat{S}}
&= \hat{H}_0 + \hat{V} + (-\hat{V} - \tfrac12\eta^2\hbar\nu\,\omega^2) + \tfrac14\eta^2\hbar\nu\,\omega^2 + O(\eta^3) \\
&= \hat{H}_0 - \frac14\eta^2\hbar\nu\,\omega^2 + O(\eta^3).
\end{align*}

\begin{equation}
\boxed{e^{\hat{S}}\hat{H}e^{-\hat{S}} = \hbar\nu\,\hat{a}^\dagger\hat{a} + \frac12\hbar\omega_\sigma\,\hat{\sigma}_z - \frac14\eta^2\hbar\nu\,\omega^2 + O(\eta^3).}
\end{equation}
\end{derivationbox}

\begin{insightbox}[与微扰论的自洽性]
BCH 展开得到的常数项 $-\frac14\eta^2\hbar\nu\,\omega^2$ 与二阶微扰能量修正 $E_{n,\sigma}^{(2)}$ 完全一致。这说明：
\begin{itemize}[nosep]
\item 到 $\eta^2$ 阶，幺正变换 $e^{\hat{S}}$ 已将微扰完全对角化；
\item 谐振子部分的能谱不变（频率仍为 $\nu$），只是整体下移一个与 $n$ 无关的常数；
\item 自旋自由度 $\sigma$ 不受影响，因为 $\hat{V}\propto\hat{\sigma}_z$ 与 $\hat{\sigma}_z$ 对易。
\end{itemize}

\textbf{更深层的联系}：$\hat{S}\propto(\hat{a}^\dagger-\hat{a})\hat{\sigma}_z$ 的生成元形式，与相干态的位移算符 $\hat{D}(\alpha)=\exp(\alpha\hat{a}^\dagger-\alpha^*\hat{a})$ 以及 BdG 的压缩算符 $\hat{S}(r)=\exp[\frac{r}{2}(\hat{a}^2-(\hat{a}^\dagger)^2)]$ 属于同一族——都是玻色算符双线性型的指数。区别在于：位移对应 $c$ 数平移（$[\hat{a},\hat{a}^\dagger]$ 闭合到 1），压缩对应双曲旋转（$\hat{a}^2$ 与 $(\hat{a}^\dagger)^2$ 的配对），而本题则是将玻色子与自旋自由度解耦的"旋转"。
\end{insightbox}
