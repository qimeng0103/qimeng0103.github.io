\section{核心模型与快速识别}

\begin{multicols}{2}

\begin{modelbox}[Bloch 定理（能带理论基础）]
\textbf{数学形式：}
\[
\psi_{n\mathbf{k}}(\mathbf{r}) = e^{i\mathbf{k}\cdot\mathbf{r}} u_{n\mathbf{k}}(\mathbf{r})
\]

\textbf{物理图像：}
\begin{itemize}
    \item $e^{i\mathbf{k}\cdot\mathbf{r}}$：自由电子行波（\en{plane wave}）
    \item $u_{n\mathbf{k}}(\mathbf{r})$：原子轨道局域波函数
    \item 周期势场导致两者"折衷"形成能带
\end{itemize}

\examnote{看到"周期势场"立即写此形式！}

\textbf{关键区分：}
\begin{itemize}
    \item \en{Crystal momentum} $\mathbf{k}$ $\neq$ 物理动量 $\mathbf{p}$
    \item 简约区方案（\en{reduced zone}）vs 扩展区方案
\end{itemize}
\end{modelbox}

\begin{tipbox}[散射问题：Born vs 分波法]
\textbf{快速决策树：}
\[
ka \gg 1 \Rightarrow \text{Born 近似} \\
ka \ll 1 \Rightarrow \text{分波法（s波主导）}
\]

\textbf{Born 近似一级公式：}
\[
f_B(\mathbf{k}',\mathbf{k}) = -\frac{m}{2\pi\hbar^2} \tilde{V}(\mathbf{q})
\]
其中 $\mathbf{q} = \mathbf{k}' - \mathbf{k}$（动量转移）

\textbf{球对称势快速公式：}
\[
\tilde{V}(q) = \frac{4\pi}{q} \int_0^\infty rV(r)\sin(qr)dr
\]

\examnote{记住这个积分形式！}
\end{tipbox}

\begin{formulabox}[紧束缚模型（\en{Tight-Binding}）]
\textbf{色散关系：}
\[
E(\mathbf{k}) = \varepsilon_0 - t \sum_{\langle i,j \rangle} e^{i\mathbf{k}\cdot(\mathbf{R}_i - \mathbf{R}_j)}
\]

\textbf{一维单带简化：}
\[
E(k) = \varepsilon_0 - 2t\cos(ka)
\]

\textbf{物理意义：}
\begin{itemize}
    \item $t$：跃迁积分（hopping），决定带宽
    \item $t > 0$：能带底部在 $k=0$
    \item 带边有效质量：$m^* = \hbar^2/(2ta^2)$
\end{itemize}
\end{formulabox}

\begin{warnbox}[常见陷阱]
\begin{itemize}
    \item 混淆 \en{reduced mass} $\mu$ 与 \en{effective mass} $m^*$
    \item 一维态密度 $g(E) \propto 1/\sqrt{E}$（发散！）
    \item 三维自由电子 $g(E) \propto \sqrt{E}$
    \item 费米能级定义：$E_F = \hbar^2 k_F^2/(2m)$（$T=0$）
\end{itemize}
\end{warnbox}

\begin{insightbox}[二次量子化本质]
\textbf{核心思想：}多体问题 $\rightarrow$ 场算符的单体问题

\textbf{产生湮灭算符：}
\[
\hat{c}_i^\dagger |...,n_i,...\rangle = \sqrt{n_i+1} |...,n_i+1,...\rangle
\]

\textbf{场算符：}
\[
\hat{\psi}(\mathbf{r}) = \sum_\alpha \phi_\alpha(\mathbf{r}) \hat{c}_\alpha
\]

\textbf{物理图像：}
粒子是场算符的激发态，类似光子在电磁场中。
\end{insightbox}

\begin{cheatbox}[快速公式备忘]
\textbf{对易关系：}
\begin{itemize}
    \item $[\hat{x}, \hat{p}] = i\hbar$
    \item $[\hat{a}, \hat{a}^\dagger] = 1$（玻色子）
    \item $\{\hat{c}_i, \hat{c}_j^\dagger\} = \delta_{ij}$（费米子）
\end{itemize}

\textbf{常用积分：}
\[
\int_{-\infty}^\infty e^{-x^2} dx = \sqrt{\pi}
\]
\[
\int_0^\infty \frac{x^{s-1}}{e^x + 1} dx = (1-2^{1-s})\Gamma(s)\zeta(s)
\]
\end{cheatbox}

\end{multicols}

\section{真题题型与解题套路}

\begin{multicols}{2}

\begin{tipbox}[题型 A：微扰论计算]
\textbf{识别特征：}
$\hat{H} = \hat{H}_0 + \lambda \hat{V}$，已知 $\hat{H}_0$ 本征态

\textbf{非简并微扰（能量到二级）：}
\begin{align*}
E_n^{(1)} &= \langle n^{(0)} | \hat{V} | n^{(0)} \rangle \\
E_n^{(2)} &= \sum_{m \neq n} \frac{|\langle m^{(0)} | \hat{V} | n^{(0)} \rangle|^2}{E_n^{(0)} - E_m^{(0)}}
\end{align*}

\textbf{简并微扰：}
在简并子空间对角化 $V_{ij} = \langle i | \hat{V} | j \rangle$

\examnote{先判断简并性！}
\end{tipbox}

\begin{tipbox}[题型 B：格林函数]
\textbf{定义速查：}
\[
G^R(\omega) = \frac{1}{\omega - H + i\eta}
\]

\textbf{Dyson 方程：}
\[
G = G_0 + G_0 \Sigma G
\]

\textbf{谱函数：}
\[
A(\mathbf{k},\omega) = -2\text{Im}\,G^R(\mathbf{k},\omega)
\]

\textbf{DOS 联系：}
\[
\rho(\omega) = \sum_\mathbf{k} A(\mathbf{k},\omega)
\]
\end{tipbox}

\begin{formulabox}[题型 C：角动量耦合]
\textbf{C-G 系数快速记忆：}
\[
|j_1 m_1\rangle |j_2 m_2\rangle = \sum_{j,m} C_{j_1 m_1, j_2 m_2}^{jm} |jm\rangle
\]

\textbf{常用：自旋-1/2 与轨道耦合}
\[
\mathbf{J} = \mathbf{L} + \mathbf{S}, \quad j = l \pm \frac{1}{2}
\]

\textbf{LS 耦合 vs jj 耦合：}
\begin{itemize}
    \item 轻原子：LS 耦合（$[\mathbf{L}, \mathbf{S}] = 0$）
    \item 重原子：jj 耦合
\end{itemize}
\end{formulabox}

\begin{warnbox}[考试中的单位陷阱]
\begin{itemize}
    \item 自然单位制：$\hbar = c = 1$（高能物理）
    \item 原子单位制：$\hbar = m_e = e = 1$
    \item 检查答案量纲！
    \item 玻尔半径：$a_0 = 4\pi\varepsilon_0\hbar^2/(m_e e^2) \approx 0.529$ Å
\end{itemize}
\end{warnbox}

\end{multicols}

\section{附录：完整推导与理解}

\begin{derivationbox}[Bloch 定理完整推导]
\textbf{出发点：}周期势场 $V(\mathbf{r} + \mathbf{R}) = V(\mathbf{r})$，其中 $\mathbf{R}$ 为晶格矢量。

\textbf{步骤 1：定义平移算符}

平移算符 $\hat{T}_\mathbf{R}$ 作用为：
\[
\hat{T}_\mathbf{R} \psi(\mathbf{r}) = \psi(\mathbf{r} + \mathbf{R})
\]

\textbf{步骤 2：证明与哈密顿量对易}

由于 $V(\mathbf{r} + \mathbf{R}) = V(\mathbf{r})$，且动能项 $-\frac{\hbar^2}{2m}\nabla^2$ 在平移下不变：
\[
[\hat{T}_\mathbf{R}, \hat{H}] = 0
\]

\textbf{步骤 3：平移算符的本征值}

由于 $\hat{T}_\mathbf{R}\hat{T}_{\mathbf{R}'} = \hat{T}_{\mathbf{R}+\mathbf{R}'}$，且 $\hat{T}_\mathbf{R}^\dagger = \hat{T}_{-\mathbf{R}}$，平移算符是么正的。

本征值必须是模为 1 的复数：
\[
\hat{T}_\mathbf{R} |\psi\rangle = c_\mathbf{R} |\psi\rangle, \quad |c_\mathbf{R}| = 1
\]

由群的性质，$c_{\mathbf{R}+\mathbf{R}'} = c_\mathbf{R} c_{\mathbf{R}'}$，这要求：
\[
c_\mathbf{R} = e^{i\mathbf{k}\cdot\mathbf{R}}
\]

\textbf{步骤 4：构造 Bloch 函数}

定义 $u_\mathbf{k}(\mathbf{r}) = e^{-i\mathbf{k}\cdot\mathbf{r}} \psi(\mathbf{r})$，验证其周期性：
\begin{align*}
u_\mathbf{k}(\mathbf{r} + \mathbf{R}) &= e^{-i\mathbf{k}\cdot(\mathbf{r}+\mathbf{R})} \psi(\mathbf{r} + \mathbf{R}) \\
&= e^{-i\mathbf{k}\cdot\mathbf{r}} e^{-i\mathbf{k}\cdot\mathbf{R}} \cdot e^{i\mathbf{k}\cdot\mathbf{R}} \psi(\mathbf{r}) \\
&= e^{-i\mathbf{k}\cdot\mathbf{r}} \psi(\mathbf{r}) = u_\mathbf{k}(\mathbf{r})
\end{align*}

\textbf{Q.E.D.}
\end{derivationbox}

\begin{derivationbox}[微扰论详细推导]
\textbf{设定：}$
\hat{H} = \hat{H}_0 + \lambda \hat{V}, \quad
\hat{H}_0 |n^{(0)}\rangle = E_n^{(0)} |n^{(0)}\rangle
$

\textbf{展开：}
\begin{align*}
E_n &= E_n^{(0)} + \lambda E_n^{(1)} + \lambda^2 E_n^{(2)} + ... \\
|n\rangle &= |n^{(0)}\rangle + \lambda |n^{(1)}\rangle + \lambda^2 |n^{(2)}\rangle + ...
\end{align*}

使用中间归一化：$\langle n^{(0)} | n \rangle = 1$，则 $\langle n^{(0)} | n^{(k)} \rangle = 0$（$k \geq 1$）。

\textbf{一级能量修正：}

将展开代入薛定谔方程，收集 $\lambda^1$ 项：
\[
\hat{H}_0 |n^{(1)}\rangle + \hat{V} |n^{(0)}\rangle = E_n^{(0)} |n^{(1)}\rangle + E_n^{(1)} |n^{(0)}\rangle
\]

左乘 $\langle n^{(0)} |$：
\[
\underbrace{\langle n^{(0)} | \hat{H}_0 | n^{(1)} \rangle}_{E_n^{(0)} \langle n^{(0)} | n^{(1)} \rangle = 0} + \langle n^{(0)} | \hat{V} | n^{(0)} \rangle = E_n^{(0)} \underbrace{\langle n^{(0)} | n^{(1)} \rangle}_{0} + E_n^{(1)}
\]

得：
\[
E_n^{(1)} = \langle n^{(0)} | \hat{V} | n^{(0)} \rangle = V_{nn}
\]

\textbf{一级态修正：}

展开 $|n^{(1)}\rangle = \sum_{m \neq n} c_m^{(1)} |m^{(0)}\rangle$，左乘 $\langle m^{(0)} |$（$m \neq n$）：
\[
E_m^{(0)} c_m^{(1)} + V_{mn} = E_n^{(0)} c_m^{(1)}
\]

解得：
\[
c_m^{(1)} = \frac{V_{mn}}{E_n^{(0)} - E_m^{(0)}}
\]

因此：
\[
|n^{(1)}\rangle = \sum_{m \neq n} \frac{V_{mn}}{E_n^{(0)} - E_m^{(0)}} |m^{(0)}\rangle
\]

\textbf{二级能量修正：}

类似地，收集 $\lambda^2$ 项并投影到 $\langle n^{(0)} |$：
\[
E_n^{(2)} = \langle n^{(0)} | \hat{V} | n^{(1)} \rangle = \sum_{m \neq n} \frac{|V_{mn}|^2}{E_n^{(0)} - E_m^{(0)}}
\]
\end{derivationbox}
