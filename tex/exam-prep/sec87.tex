\section{固体理论：各向异性有效质量的回旋共振}
\label{sec:anisotropic-cyclotron}

\begin{modelbox}[题目：椭球等能面的回旋共振有效质量]
设电子等能面为椭球
\[
E(\bm{k})=\frac{\hbar^2k_x^2}{2m_1^*}+\frac{\hbar^2k_y^2}{2m_2^*}+\frac{\hbar^2k_z^2}{2m_3^*},
\]
其中 $k_x,k_y,k_z$ 是相互垂直的主轴。若外加磁场相对于椭球主轴的方向余弦分别为 $\cos\alpha,\cos\beta,\cos\gamma$，证明电子绕磁场回旋共振频率为
\[
\omega_c=\frac{eB}{m^*},\qquad
\frac{1}{m^*}=\sqrt{\frac{m_1^*\cos^2\alpha+m_2^*\cos^2\beta+m_3^*\cos^2\gamma}{m_1^*m_2^*m_3^*}},
\]
其中 $m^*$ 称为电子回旋共振频率有效质量。
\end{modelbox}

\begin{tipbox}[考点快速分析]
\begin{itemize}
    \item \textbf{标度变换}：将各向异性有效质量问题映射为各向同性问题
    \item \textbf{坐标变换}：$x_i'=\sqrt{m_i^*/m_0}\,x_i$，使哈密顿量化为自由电子形式
    \item \textbf{磁场变换}：$B_i'=(\sqrt{m_j^*m_k^*}/m_0)\,B_i$（$i,j,k$ 轮换）
    \item \textbf{各向异性效应}：$m^*$ 依赖于磁场方向与椭球主轴的相对取向
\end{itemize}
\end{tipbox}

\begin{derivationbox}[标准解题过程]
\textbf{Step 1: 有效质量方程与标度变换}

有效质量张量对角化后，哈密顿量为
\begin{equation}
H=\sum_{i=1}^3\frac{1}{2m_i^*}\Bigl(-i\hbar\frac{\partial}{\partial x_i}+eA_i\Bigr)^2.
\end{equation}

引入坐标和矢势的\textbf{标度变换}（$m_0$ 为参考质量，不影响最终结果）：
\begin{equation}
x_i'=\sqrt{\frac{m_i^*}{m_0}}\,x_i,\qquad A_i'=\sqrt{\frac{m_i^*}{m_0}}\,A_i.
\end{equation}

则 $\partial/\partial x_i=\sqrt{m_i^*/m_0}\,\partial/\partial x_i'$，第 $i$ 项变为
\begin{equation}
\frac{1}{2m_i^*}\Bigl(\sqrt{\frac{m_i^*}{m_0}}(-i\hbar\partial_{x_i'})+e\sqrt{\frac{m_0}{m_i^*}}A_i'\Bigr)^2
=\frac{1}{2m_0}(-i\hbar\partial_{x_i'}+eA_i')^2.
\end{equation}

总和化为各向同性形式
\begin{equation}
H=\frac{1}{2m_0}(-i\hbar\nabla'+e\bm{A}')^2.
\end{equation}

\textbf{Step 2: 变换后的磁场}

原磁场 $\bm{B}=\nabla\times\bm{A}$。利用变换关系反推，变换后的磁场分量满足
\begin{equation}
B_1'=\frac{\sqrt{m_2^*m_3^*}}{m_0}\,B_1,\quad
B_2'=\frac{\sqrt{m_1^*m_3^*}}{m_0}\,B_2,\quad
B_3'=\frac{\sqrt{m_1^*m_2^*}}{m_0}\,B_3.
\end{equation}

\textbf{Step 3: 回旋频率与有效质量}

在新坐标系中，电子等效为质量 $m_0$ 的自由电子，感受到磁场 $B'=|\bm{B}'|$。由均匀磁场中的回旋频率公式
\begin{equation}
\omega_c=\frac{eB'}{m_0}=\frac{e}{m_0}\sqrt{B_1'^2+B_2'^2+B_3'^2}.
\end{equation}

代入 $B_i'$ 的分量：
\begin{equation}
\omega_c=\frac{e}{m_0}\sqrt{\frac{m_0^2}{m_2^*m_3^*}B_1^2+\frac{m_0^2}{m_1^*m_3^*}B_2^2+\frac{m_0^2}{m_1^*m_2^*}B_3^2}
=eB\sqrt{\frac{\cos^2\alpha}{m_2^*m_3^*}+\frac{\cos^2\beta}{m_1^*m_3^*}+\frac{\cos^2\gamma}{m_1^*m_2^*}}.
\end{equation}

提取公分母并整理：
\begin{equation}
\omega_c=eB\sqrt{\frac{m_1^*\cos^2\alpha+m_2^*\cos^2\beta+m_3^*\cos^2\gamma}{m_1^*m_2^*m_3^*}}.
\end{equation}

写成 $\omega_c=eB/m^*$ 的形式，即得
\begin{equation}
\boxed{\frac{1}{m^*}=\sqrt{\frac{m_1^*\cos^2\alpha+m_2^*\cos^2\beta+m_3^*\cos^2\gamma}{m_1^*m_2^*m_3^*}}.}
\end{equation}
\end{derivationbox}

\begin{insightbox}[物理图像]
\begin{itemize}
    \item 标度变换的物理本质：将椭球等能面``拉伸''为球面，使问题回到熟悉的自由电子情形。变换后的坐标系中空间不再是欧几里得的，但哈密顿量形式简单。
    \item $m^*$ 依赖于 $(\alpha,\beta,\gamma)$，即磁场方向。这是实验测定有效质量张量各向异性的基础——只需改变磁场方向、测量共振频率的角依赖，即可反推出 $m_1^*,m_2^*,m_3^*$。
    \item 若 $m_1^*=m_2^*=m_3^*=m$，则 $1/m^*=1/m$，退化为各向同性结果，与 sec86 一致。
\end{itemize}
\end{insightbox}
