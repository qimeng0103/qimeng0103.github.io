\section{固体理论：一维双原子链的热容——$M\gg m$极限}
\label{sec:1d-diatomic-heat-capacity}

\begin{modelbox}[题目：一维复式格子的晶格热容量]
考虑由两种不同原子（质量分别为 $M$ 和 $m$）所构成的一维复式格子，相邻同种原子间距离为 $2a$，相互作用恢复力常数为 $\beta$。设 $M\gg m$，求晶格热容量，并讨论高低温极限情况。
\end{modelbox}

\begin{tipbox}[考点快速分析]
\begin{itemize}
    \item \textbf{一维双原子链色散}：两支格波——声学支（同相振动）和光学支（反相振动）
    \item \textbf{$M\gg m$ 近似}：光学支频率几乎与 $q$ 无关（爱因斯坦模型）；声学支等效为轻原子静止、重原子振动的单原子链
    \item \textbf{一维模式密度}：$D(\omega)\propto 1/\sqrt{\omega_0^2-\omega^2}$（范霍夫奇点）
    \item \textbf{高温极限}：杜隆-珀蒂定律，每原子 $k_B$，共 $2N$ 个原子 $\Rightarrow C_V\to2Nk_B$
    \item \textbf{低温极限}：光学支指数冻结，声学支主导，$C_V\propto T$（一维特征，区别于三维德拜 $T^3$ 律）
\end{itemize}
\end{tipbox}

\begin{derivationbox}[标准解题过程]
\textbf{Step 0: 色散关系的推导与近似}

\textit{运动方程的建立}：一维双原子链中，相邻异种原子间用弹簧（力常数 $\beta$）连接。设第 $2n+1$ 个原子（质量 $m$）位移为 $x_{2n+1}$，第 $2n+2$ 个原子（质量 $M$）位移为 $x_{2n+2}$。每个原子只受左右两个近邻弹簧的作用：
\begin{align}
m\ddot{x}_{2n+1}&=\beta(x_{2n+2}-x_{2n+1})-\beta(x_{2n+1}-x_{2n})\nonumber\\
&=\beta(x_{2n}+x_{2n+2}-2x_{2n+1}),\\
M\ddot{x}_{2n+2}&=\beta(x_{2n+3}-x_{2n+2})-\beta(x_{2n+2}-x_{2n+1})\nonumber\\
&=\beta(x_{2n+1}+x_{2n+3}-2x_{2n+2}).
\end{align}

\textit{试探解与久期方程}：设格波解
\begin{equation}
x_{2n+1}=Ae^{i[q(2n+1)a-\omega t]},\qquad x_{2n+2}=Be^{i[q(2n+2)a-\omega t]}.
\end{equation}

代入运动方程，利用 $x_{2n}=Ae^{i[q\cdot2na-\omega t]}=Ae^{-iqa}e^{i[q(2n+1)a-\omega t]}$ 等关系，约去公共相位因子后得到关于 $A,B$ 的齐次方程组：
\begin{align}
(2\beta-m\omega^2)A-2\beta\cos(qa)\,B&=0,\\
-2\beta\cos(qa)\,A+(2\beta-M\omega^2)B&=0.
\end{align}

有非零解的条件是系数行列式为零：
\begin{equation}
\begin{vmatrix}
2\beta-m\omega^2 & -2\beta\cos(qa)\\
-2\beta\cos(qa) & 2\beta-M\omega^2
\end{vmatrix}=0.
\end{equation}

展开即得色散方程：
\begin{equation}
mM\omega^4-2\beta(m+M)\omega^2+4\beta^2\sin^2(qa)=0.
\end{equation}

解得两支格波：
\begin{equation}
\omega_{\pm}^2=\frac{\beta}{mM}\Bigl[(m+M)\pm\sqrt{(m+M)^2-4mM\sin^2(qa)}\Bigr].
\label{eq:diatomic-dispersion-sec54}
\end{equation}

\textit{光学支}（$\omega_+$，取 $+$ 号）：在 $M\gg m$ 时，$m/M\ll1$，对 \eqref{eq:diatomic-dispersion-sec54} 中根号内展开：
\begin{equation}
\sqrt{(m+M)^2-4mM\sin^2(qa)}=(m+M)\sqrt{1-\frac{4mM}{(m+M)^2}\sin^2(qa)}.
\end{equation}

当 $m/M\ll1$ 时，$4mM/(m+M)^2\approx4m/M\ll1$，根号 $\to 1$（与 $q$ 的依赖为 $O(m/M)$ 高阶小量），故
\begin{equation}
\omega_+^2\approx\frac{\beta}{mM}\Bigl[(m+M)+(m+M)\Bigr]=\frac{2\beta(m+M)}{mM}\approx\frac{2\beta}{m}\equiv\omega_0^2.
\end{equation}

光学支近似为与 $q$ 无关的常数，可用\textbf{爱因斯坦模型}处理。

\textit{声学支}（$\omega_-$，取 $-$ 号）：同样取 $M\gg m$，此时根号前为负号：
\begin{equation}
\omega_-^2\approx\frac{\beta}{mM}\Bigl[(m+M)-(m+M)\bigl(1-\tfrac{2mM}{(m+M)^2}\sin^2(qa)\bigr)\Bigr]
=\frac{2\beta}{m+M}\sin^2(qa)\approx\frac{2\beta}{M}\sin^2(qa).
\end{equation}

定义 $\omega_0'=\sqrt{2\beta/M}$，则
\begin{equation}
\omega_-=\omega_0'|\sin(qa)|.
\end{equation}

这正是轻原子 $m$ 静止、重原子 $M$ 构成的有效单原子链的色散关系。

\textbf{Step 1: 光学支对热容量的贡献（爱因斯坦模型）}

光学支共 $N$ 个模式（$N$ 为原胞数），每个模式频率均为 $\omega_0$。总振动能：
\begin{equation}
E_+=\sum_{\text{光学支}}\frac{\hbar\omega_0}{e^{\hbar\omega_0/k_BT}-1}
=\frac{N\hbar\omega_0}{e^{\hbar\omega_0/k_BT}-1}.
\end{equation}

热容量：
\begin{equation}
C_V^+=\frac{\partial E_+}{\partial T}
=Nk_B\Bigl(\frac{\hbar\omega_0}{k_BT}\Bigr)^2\frac{e^{\hbar\omega_0/k_BT}}{\bigl(e^{\hbar\omega_0/k_BT}-1\bigr)^2}
=Nk_B\Bigl(\frac{\Theta_E}{T}\Bigr)^2\frac{e^{\Theta_E/T}}{\bigl(e^{\Theta_E/T}-1\bigr)^2},
\end{equation}
其中爱因斯坦温度 $\Theta_E=\hbar\omega_0/k_B$。

\textbf{Step 2: 声学支对热容量的贡献}

\textit{模式密度}：声学支色散 $\omega=\omega_0'|\sin(qa)|$。

一维链长 $L=N\cdot2a$（$N$ 个原胞，每胞长度 $2a$）。波矢 $q$ 在第一布里渊区内取值，间距 $\Delta q=2\pi/L$，故单位 $q$ 空间长度内的模式数为
\begin{equation}
\frac{1}{\Delta q}=\frac{L}{2\pi}=\frac{Na}{\pi}.
\end{equation}

考虑 $q$ 的正负两支（$+q$ 和 $-q$ 对应同一频率），频率区间 $d\omega$ 内的模式数为
\begin{equation}
D(\omega)\,d\omega=2\cdot\frac{L}{2\pi}\,dq=\frac{Na}{\pi}\,dq.
\end{equation}

由色散关系求 $dq/d\omega$：
\begin{equation}
\omega=\omega_0'\sin(qa)\quad\Longrightarrow\quad
\frac{d\omega}{dq}=\omega_0'a\cos(qa)=\omega_0'a\sqrt{1-\sin^2(qa)}=a\sqrt{\omega_0'^2-\omega^2}.
\end{equation}

因此
\begin{equation}
D(\omega)=\frac{Na}{\pi}\cdot\frac{1}{d\omega/dq}
=\frac{Na}{\pi}\cdot\frac{1}{a\sqrt{\omega_0'^2-\omega^2}}
=\frac{2N}{\pi}\frac{1}{\sqrt{\omega_0'^2-\omega^2}},\qquad 0\le\omega\le\omega_0'.
\label{eq:1d-Domega}
\end{equation}

\textit{验证}：总模式数应为 $N$（一维单原子链 $N$ 个原胞，$N$ 个声学模）：
\begin{equation}
\int_0^{\omega_0'}D(\omega)\,d\omega
=\frac{2N}{\pi}\int_0^{\omega_0'}\frac{d\omega}{\sqrt{\omega_0'^2-\omega^2}}
=\frac{2N}{\pi}\cdot\frac{\pi}{2}=N.\quad\checkmark
\end{equation}

\textit{热容量}：声学支总振动能
\begin{equation}
E_-=\int_0^{\omega_0'}\frac{\hbar\omega}{e^{\hbar\omega/k_BT}-1}\,D(\omega)\,d\omega.
\end{equation}

故
\begin{equation}
C_V^-=k_B\int_0^{\omega_0'}\Bigl(\frac{\hbar\omega}{k_BT}\Bigr)^2\frac{e^{\hbar\omega/k_BT}}{\bigl(e^{\hbar\omega/k_BT}-1\bigr)^2}\,D(\omega)\,d\omega.
\end{equation}

代入 \eqref{eq:1d-Domega}：
\begin{equation}
C_V^-=\frac{2Nk_B}{\pi}\int_0^{\omega_0'}\frac{1}{\sqrt{\omega_0'^2-\omega^2}}\Bigl(\frac{\hbar\omega}{k_BT}\Bigr)^2\frac{e^{\hbar\omega/k_BT}}{\bigl(e^{\hbar\omega/k_BT}-1\bigr)^2}\,d\omega.
\end{equation}

\textbf{Step 3: 高温极限 ($T\gg\Theta_E,\Theta_D$)}

\textit{光学支}：$e^{\Theta_E/T}-1\approx\Theta_E/T$，
\begin{equation}
C_V^+\approx Nk_B\Bigl(\frac{\Theta_E}{T}\Bigr)^2\frac{1}{(\Theta_E/T)^2}=Nk_B.
\end{equation}

\textit{声学支}：$e^{\hbar\omega/k_BT}-1\approx\hbar\omega/k_BT$，被积函数化简为
\begin{equation}
\Bigl(\frac{\hbar\omega}{k_BT}\Bigr)^2\frac{1}{(\hbar\omega/k_BT)^2}\cdot\frac{1}{\sqrt{\omega_0'^2-\omega^2}}=\frac{1}{\sqrt{\omega_0'^2-\omega^2}}.
\end{equation}

故
\begin{equation}
C_V^-\approx\frac{2Nk_B}{\pi}\int_0^{\omega_0'}\frac{d\omega}{\sqrt{\omega_0'^2-\omega^2}}
=\frac{2Nk_B}{\pi}\cdot\frac{\pi}{2}=Nk_B.
\end{equation}

\textbf{高温总热容量}：
\begin{equation}
C_V=C_V^++C_V^-\approx Nk_B+Nk_B=2Nk_B.
\end{equation}
符合杜隆-珀蒂定律：共 $2N$ 个原子，每个原子贡献 $k_B$。

\textbf{Step 4: 低温极限 ($T\ll\Theta_E,\Theta_D$)}

\textit{光学支}：$\Theta_E/T\gg1$，$e^{\Theta_E/T}\gg1$，
\begin{equation}
C_V^+\approx Nk_B\Bigl(\frac{\Theta_E}{T}\Bigr)^2e^{-\Theta_E/T}\to0.
\end{equation}
光学支被指数冻结，贡献可忽略。

\textit{声学支}：低温下只有 $\omega\ll\omega_0'$ 的低频模式被激发，可作两个近似：
\begin{itemize}
    \item 分母 $\sqrt{\omega_0'^2-\omega^2}\approx\omega_0'$（常数），因主要贡献来自 $\omega\sim k_BT/\hbar\ll\omega_0'$
    \item 积分上限延拓至 $\infty$（因玻色因子在 $\omega\gtrsim k_BT/\hbar$ 处指数截断，$\omega_0'$ 处无贡献）
\end{itemize}

于是
\begin{equation}
C_V^-\approx\frac{2Nk_B}{\pi\omega_0'}\int_0^{\infty}\Bigl(\frac{\hbar\omega}{k_BT}\Bigr)^2\frac{e^{\hbar\omega/k_BT}}{\bigl(e^{\hbar\omega/k_BT}-1\bigr)^2}\,d\omega.
\end{equation}

令 $x=\hbar\omega/k_BT$，则 $\hbar\omega/k_BT=x$，$d\omega=k_BT\,dx/\hbar$：
\begin{align}
C_V^-&\approx\frac{2Nk_B}{\pi\omega_0'}\int_0^{\infty}x^2\frac{e^x}{(e^x-1)^2}\cdot\frac{k_BT}{\hbar}\,dx\\
&=\frac{2Nk_B}{\pi\omega_0'}\cdot\frac{k_BT}{\hbar}\int_0^{\infty}\frac{x^2e^x}{(e^x-1)^2}\,dx.
\end{align}

利用标准积分 $\displaystyle\int_0^{\infty}\frac{x^2e^x}{(e^x-1)^2}\,dx=2\zeta(2)=\frac{\pi^2}{3}$，得
\begin{equation}
C_V^-\approx\frac{2Nk_B}{\pi\omega_0'}\cdot\frac{k_BT}{\hbar}\cdot\frac{\pi^2}{3}
=\frac{2\pi Nk_B^2T}{3\hbar\omega_0'}\propto T.
\end{equation}

\textbf{低温总热容量}：
\begin{equation}
C_V=C_V^++C_V^-\approx0+\frac{2\pi Nk_B^2T}{3\hbar\omega_0'}\propto T.
\end{equation}
\end{derivationbox}

\begin{insightbox}[物理图像与概念深化]
\textbf{1. 为什么光学支能用爱因斯坦模型？}

$M\gg m$ 意味着重原子几乎不动，轻原子在``固定''的势能阱中振动。光学支描述的是轻原子相对于重原子的反相振动，其频率主要由轻原子质量和恢复力常数决定，与相邻原胞的相位关系（$q$）很弱——就像一个被弹簧连在墙上的小球，弹簧另一端固定。这就是爱因斯坦模型的物理：$N$ 个独立的、频率相同的谐振子。

\textbf{2. 为什么声学支等效为单原子链？}

声学支描述的是同相振动（长波极限下两个子晶格一起动）。由于 $M\gg m$，轻原子 $m$ 像``寄生虫''一样跟着重原子 $M$ 一起运动，其惯性贡献可忽略。有效振动质量就是 $M$，晶格常数就是 $2a$（同种原子间距），于是退化为单原子链。

\textbf{3. 一维模式密度的范霍夫奇点}

\eqref{eq:1d-Domega} 中 $D(\omega)\propto1/\sqrt{\omega_0'^2-\omega^2}$ 在 $\omega=\omega_0'$ 处发散，这是\textbf{一维范霍夫奇点}（Van Hove singularity）的典型特征。

\textit{群速度为零与态密度堆积的物理联系}：

态密度的定义是``单位频率区间内的状态数''。从波矢空间到频率空间的变换中：
\begin{equation}
D(\omega)=\frac{dn}{d\omega}=\underbrace{\frac{dn}{dq}}_{\text{常数}}\cdot\underbrace{\frac{dq}{d\omega}}_{=1/v_g}
\propto\frac{1}{|v_g|}=\frac{1}{|d\omega/dq|}.
\end{equation}

\begin{itemize}
    \item $dn/dq$ 是波矢空间的态密度，一维中均匀分布，为常数
    \item $dq/d\omega=1/v_g$ 是\textbf{关键因子}：群速度越小，频率空间中``挤''的状态越多
\end{itemize}

\textit{直观图像}：想象一条传送带以速度 $v_g$ 运送箱子（状态）。
\begin{itemize}
    \item 正常速度时：箱子均匀分布，单位长度内箱子数正常
    \item 速度突然变慢：箱子在此处堆积，单位长度内箱子数剧增
    \item 速度趋于零：箱子几乎不动，堆积成山，密度趋于无穷
\end{itemize}

在布里渊区边界 $q=\pi/(2a)$ 处，声学支色散 $\omega=\omega_0'|\sin(qa)|$ 达到极大值，曲线平坦：
\begin{equation}
\frac{d\omega}{dq}=\omega_0'a\cos(qa)\big|_{q=\pi/(2a)}=0.
\end{equation}

\textit{为什么群速度在这里为零？} 布里渊区边界处相邻原胞相位差为 $\pi$（反相振动）。每个原子同时受到左右两个近邻的恢复力，由于位移对称，这两个力恰好大小相等、方向相反，合力为零——原子``不知道该往哪边走''，能量无法传播，形成驻波。大量模式集中在这个频率附近，导致态密度发散。

\textit{奇点可积}：虽然 $D(\omega)\to\infty$，但 $D(\omega)$ 的积分（总模式数）是有限的：$\int_0^{\omega_0'}D(\omega)\,d\omega=N$。在热容积分中，被积函数还包含玻色因子 $n(\omega)$，热容量仍然是有限值。

\textbf{4. 为什么低温下 $C_V\propto T$ 而非 $T^3$？}

\begin{center}
\begin{tabular}{ccc}
\toprule
维度 & 模式密度 & 低温热容 \\
\midrule
一维 & $D(\omega)\propto\omega^0$（常数） & $C_V\propto T$ \\
二维 & $D(\omega)\propto\omega$ & $C_V\propto T^2$ \\
三维 & $D(\omega)\propto\omega^2$ & $C_V\propto T^3$ \\
\bottomrule
\end{tabular}
\end{center}

低温下有效频率范围 $\omega_{\text{cutoff}}\sim k_BT/\hbar$。一维中 $D(\omega)$ 在此区间积分得 $N_{\text{modes}}\propto T^1$（而非三维的 $T^3$），每个模式能量 $\sim k_BT$，故 $U\propto T^2$，$C_V=dU/dT\propto T$。

\textbf{5. 杜隆-珀蒂定律的适用条件}

高温 $C_V\to2Nk_B$ 意味着每个振动自由度贡献 $k_B$。一维双原子链每个原胞有 2 个原子，每个原子 1 个自由度，共 $2N$ 个简正模，故总热容为 $2Nk_B$。这与经典能量均分定理一致：每个谐振子平均能量 $k_BT$（动能 $k_BT/2$ + 势能 $k_BT/2$）。
\end{insightbox}

\begin{thinkbox}[考场速记：一维双原子链热容问题的四步法]
遇到``一维双原子链求热容''类问题：
\begin{enumerate}
    \item \textbf{写色散并取近似}：$M\gg m$ 时光学支 $\omega_+\approx\text{const}$（爱因斯坦），声学支 $\omega_-\approx\omega_0'|\sin(qa)|$（德拜/单原子链）
    \item \textbf{分别求两支热容}：光学支直接爱因斯坦公式；声学支先求 $D(\omega)$ 再积分
    \item \textbf{高温展开}：$e^x-1\approx x$，验证杜隆-珀蒂 $C_V\to2Nk_B$
    \item \textbf{低温近似}：光学支指数冻结；声学支低频近似 $D(\omega)\approx\text{const}$，得 $C_V\propto T$
\end{enumerate}
\textit{核心记忆}：$M\gg m$ = 光学支（轻原子独立振动）+ 声学支（重原子单原子链）。一维低温热容永远是 $\propto T$，不是 $T^3$。
\end{thinkbox}
