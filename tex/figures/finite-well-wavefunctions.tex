\documentclass[tikz,border=5pt]{standalone}
\usepackage{tikz}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}

\begin{document}

% Figure 1: Finite Square Well Potential and Wave Functions
\begin{tikzpicture}[scale=1.2]

% Left panel: Potential
\begin{scope}[shift={(-4,0)}]
    % Axes
    \draw[->] (-2.5,0) -- (2.5,0) node[right] {$x/a$};
    \draw[->] (0,-2.5) -- (0,0.8) node[above] {$V(x)/V_0$};
    
    % Potential well
    \draw[thick,blue] (-2,0) -- (-1,0) -- (-1,-2) -- (1,-2) -- (1,0) -- (2,0);
    
    % Energy levels
    \draw[red,dashed] (-1.8,-1.5) -- (1.8,-1.5) node[right,font=\small] {$E_0$};
    \draw[orange,dashed] (-1.8,-0.5) -- (1.8,-0.5) node[right,font=\small] {$E_1$};
    
    % Labels
    \node[font=\small] at (0,-2.3) {$-V_0$};
    \node[font=\small] at (-1.4,0.2) {$-a$};
    \node[font=\small] at (1.4,0.2) {$a$};
    \node[font=\bfseries] at (0,0.5) {Potential Well};
\end{scope}

% Right panel: Wave functions
\begin{scope}[shift={(3,0)}]
    % Axes
    \draw[->] (-2.5,0) -- (2.5,0) node[right] {$x/a$};
    \draw[->] (0,-1.5) -- (0,2) node[above] {$\psi(x)$};
    
    % Well boundaries
    \draw[gray,dotted] (-1,-1.5) -- (-1,2);
    \draw[gray,dotted] (1,-1.5) -- (1,2);
    
    % Even parity wavefunction (ground state)
    \draw[thick,red,samples=100,domain=-2.5:2.5,smooth] 
        plot (\x, {(\x > -1 && \x < 1) ? 1.2*cos(deg(pi*\x/2*0.8)) : 
                   (\x <= -1) ? 0.76*exp(1.2*(\x+1)) : 
                   0.76*exp(-1.2*(\x-1))});
    \node[red,font=\small] at (2,1) {$\psi_0$ (even)};
    
    % Odd parity wavefunction (first excited)
    \draw[thick,blue,samples=100,domain=-2.5:2.5,smooth] 
        plot (\x, {(\x > -1 && \x < 1) ? 1.5*sin(deg(pi*\x)) : 
                   (\x <= -1) ? -0.5*exp(1.2*(\x+1)) : 
                   0.5*exp(-1.2*(\x-1))});
    \node[blue,font=\small] at (2,-0.8) {$\psi_1$ (odd)};
    
    \node[font=\bfseries] at (0,2.3) {Bound State Wave Functions};
    \node[font=\small] at (-1.4,-1.3) {$-a$};
    \node[font=\small] at (1.4,-1.3) {$a$};
\end{scope}

\end{tikzpicture}

\end{document}
